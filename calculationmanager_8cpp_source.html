88void CalculationManager::loadAlgorithms()
89{
90 std::vector<std::string> search_dirs = {
91 "algorithms",
92 "../algorithms"
93 };
94
95#if defined(EPT_ALGORITHMS_DIR)
96 // add ept algorithms install dir
97 search_dirs.push_back(EPT_ALGORITHMS_DIR);
98#endif
99#if defined(_WIN32)
100# if WINAPI_FAMILY_PARTITION(WINAPI_PARTITION_DESKTOP)
101 // add current executable dir (required when starting with wrong working dir)
102 char filename[MAX_PATH];
103 if (GetModuleFileNameA(NULL, filename, MAX_PATH)) {
104 // remove name of exe to get dir
105 std::string dir = filename;
106 dir = dir.substr(0, dir.rfind('\\'));
107 search_dirs.push_back(dir + "\\algorithms");
108 } else {
109 LogW("Executable dir could not be added to the algorithm search path: Error in GetModuleFileName");
110 }
111# endif
112#elif defined(__unix)
113 // add system directories on unix systems
114 search_dirs.push_back("/usr/lib/entropypianotuner/algorithms");
115 search_dirs.push_back("/usr/lib64/entropypianotuner/algorithms");
116 search_dirs.push_back("~/.entropypianotuner/algorithms");
117
118
119 // add current executable dir (required when starting with wrong working dir)
120 char result[PATH_MAX];
121 ssize_t count = readlink("/proc/self/exe", result, PATH_MAX);
122 const char *path;
123 if (count != -1) {
124 path = dirname(result);
125 }
126 search_dirs.push_back(std::string(path) + "/algorithms");
127#endif
128 loadAlgorithms(search_dirs);
129
130}

132void CalculationManager::loadAlgorithms(const std::vector<std::string> &algorithmsDirs)
133{
134#ifdef EPT_NO_SHARED_ALGORITHMS
135 (void)algorithmsDirs; // empty function
136#else
137
138 auto is_library_ending = [](const std::string &str, const std::string &type) {
139 const size_t pos = str.find(type);
140 // file does not contain file ending
141 if (pos == std::string::npos) {return false;}
142
143 // library ends without version (not required)
144 if (str.size() - pos == type.size()) {return true;}
145
146 // check if end matches a version
147 const std::string ending = str.substr(pos);
148 const std::string regex_str = type + "(\\.[0-9]+){1,3}";
149 std::regex reg(regex_str);
150 std::smatch pieces_match;
151
152 return std::regex_match(ending, pieces_match, reg);
153 };
154
155 auto is_library_type = [is_library_ending](const std::string &str, const std::string &suffix) {
156 if (is_library_ending(str, suffix)) {return true;}
157
158 return false;
159 };
160
161 auto is_library = [is_library_type](const std::string &str) {
162 return is_library_type(str, ".so") || is_library_type(str, ".dll") || is_library_type(str, ".dylib");
163 };
164
165 auto parse_version = [](const std::string &str, int &major, int &minor, int &patch) {
166 try {
167 std::array<int *, 3> v_out = {{&major, &minor, &patch}};
168
169 const std::string regex_str = "(([0-9]+)\\.)?(([0-9]+)\\.)?([0-9]+)";
170 std::regex reg(regex_str);
171 std::smatch pieces_match;
172
173 EptAssert(std::regex_match(str, pieces_match, reg), "RegEx must match.");
174
175 int out_index = 0;
176 for (size_t i = 1; i < pieces_match.size(); ++i) {
177 std::string m = pieces_match[i];
178 if (m.find(".") != std::string::npos) {continue;}
179
180 *v_out[out_index] = std::atoi(m.c_str());
181 ++out_index;
182 }
183 } catch (const std::regex_error &e) {
184 LogE("Regex Error (code {}): {}", static_cast<int>(e.code()), e.what());
185 }
186 };
187
188 for (auto algorithmsDir : algorithmsDirs) {
189 DIR *dir = opendir(algorithmsDir.c_str());
190 struct dirent *ent;
191 if (!dir) {
192 LogI("Algorithm plugins directory at '%s' could not be opened.", algorithmsDir.c_str());
193 continue;
194 } else {
195 LogI("Loading algorithms from '%s'", algorithmsDir.c_str());
196 }
197 while ((ent = readdir(dir)) != nullptr) {
198 const std::string filename = algorithmsDir + "/" + std::string(ent->d_name);
199 if (is_library(filename)) {
200 LogI("Reading algorithm %s", filename.c_str());
201
202 SharedLibraryPtr lib(new SharedLibrary(filename));
203 if (lib->is_open()) {
204
205 // Get plugin descriptor and exports
206 AlgorithmPluginDetails* info;
207 if (!lib->sym("exports", reinterpret_cast<void**>(&info))) {
208 LogW("Symbol lookup error.");
209 continue;
210 }
211
212 // loaded info
213 LogI("Algorithm Info: API Version = %i, "
214 "File Name = %s, "
215 "Class Name = %s, "
216 "Plugin Name = %s, "
217 "Plugin Version = %s ",
218 info->apiVersion,
219 info->fileName,
220 info->className,
221 info->description.getAlgorithmName().c_str(),
222 info->description.getVersion().c_str());
223
224 // API Version checking
225 if (info->apiVersion != ALGORITHM_PLUGIN_API_VERSION) {
226 LogW("Plugin ABI version mismatch. Expected %i, got %i.",
227 ALGORITHM_PLUGIN_API_VERSION, info->apiVersion);
228 continue;
229 }
230
231 int cmajor = 0, cminor = 0, cpatch = 0;
232 int nmajor = 0, nminor = 0, npatch = 0;
233 // check if algorithms already exists
234 if (hasAlgorithm(info->description.getAlgorithmName())) {
235 // check version
236 parse_version(info->description.getVersion(), nmajor, nminor, npatch);
237 parse_version(getAlgorithmFactoryDescriptionById(info->description.getAlgorithmName()).getVersion(), cmajor, cminor, cpatch);
238
239 if (nmajor > cmajor || (nmajor == cmajor && nminor > cminor) || (nmajor == cmajor && nminor == cminor && npatch > cpatch)) {
240 LogI("Overwriting algorithm with newer version");
241 } else {
242 LogI("Algorithm already present, skipping.");
243 continue;
244 }
245 }
246
247 // Instantiate the plugin
248 // Upon first call the AlgorithmFactory will register itself in the constructor
249 // to the CalculationManager
250 AlgorithmFactoryBase *factory = reinterpret_cast<AlgorithmFactoryBase*>(info->factoryFunc());
251
252 // store lib
253 this->registerFactory(factory->getDescription().getAlgorithmName(), factory);
254 mLoadedAlgorithmLibraries.push_back(lib);
255 } else {
256 LogW("Algorithm could not be added.");
257 }
258 }
259 }
260
261 closedir(dir);
262 }
263#endif
264}

309void CalculationManager::registerFactory(const std::string &name, AlgorithmFactoryBase* factory)
310{
311 if (mAlgorithms.count(name) == 1) {
312 EPT_EXCEPT(EptException::ERR_DUPLICATE_ITEM, "An algorithm with name '" + name + "' already exists.");
313 }
314
315 mAlgorithms[name] = factory;
