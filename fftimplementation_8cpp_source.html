134//-----------------------------------------------------------------------------
135// Private function: construct a plan for transformations real->complex
136//-----------------------------------------------------------------------------
137
150
151void FFT_Implementation::updatePlan (const FFTRealVector &in,
152 unsigned flags)
153{
154 // if plan still exists and input vector has the same size do nothing
155 if (mPlanRC and mRvec1 and mCvec2 and in.size()==mNRC) return
156 EptAssert(in.size()>0,"vector size has to be nonzero");
157
158 std::lock_guard<std::mutex> lock(mPlanMutex);
159 try {
160 // delete old plan and vectors
161 if (mPlanRC) fftw_destroy_plan(mPlanRC);
162 if (mRvec1) free(mRvec1);
163 if (mCvec2) fftw_free(mCvec2);
164
165 // allocate new vectors and create a new plan
166 mNRC = in.size();
167 mRvec1 = static_cast<double *>(malloc(mNRC*sizeof(double)));
168 mCvec2 = static_cast<fftw_complex*>(fftw_malloc((mNRC/2+1)*sizeof(fftw_complex)));
169 EptAssert(mRvec1, "May not be nullptr");
170 EptAssert(mCvec2, "May not be nullptr");
171 mPlanRC = fftw_plan_dft_r2c_1d (static_cast<int>(mNRC), mRvec1, mCvec2, flags);
172 }
173 catch (...) {
174 LogE("fftw_pplan_dft_r2c_1d throwed an exception");
175 }
176}

195
196void FFT_Implementation::updatePlan (const FFTComplexVector &in,
197 unsigned flags)
198{
199 // if plan still exists and input vector has the same size do nothing
200 if (mPlanCR and mCvec1 and mRvec2 and in.size()==mNCR/2+1) return;
201 EptAssert(in.size()>0,"vector size has to be nonzero");
202
203 std::lock_guard<std::mutex> lock(mPlanMutex);
204 try {
205 // delete old plan and vectors
206 if (mPlanCR) fftw_destroy_plan(mPlanCR);
207 if (mCvec1) fftw_free(mCvec1);
208 if (mRvec2) free(mRvec2);
209
210 // allocate new vectors and create a new plan
211 mNCR = 2*in.size()-2;
212 mCvec1 = static_cast<fftw_complex*>(fftw_malloc((mNCR/2+1)*sizeof(fftw_complex)));
213 mRvec2 = static_cast<double *>(malloc(mNCR*sizeof(double)));
214 EptAssert(mCvec1, "May not be nullptr");
215 EptAssert(mRvec2, "May not be nullptr");
216 mPlanCR = fftw_plan_dft_c2r_1d (static_cast<int>(mNCR), mCvec1, mRvec2, flags);
217 }
218 catch (...) {
219 LogE("fftw_pplan_dft_c2r_1d throwed an exception");
220 }
221}

224//-----------------------------------------------------------------------------
225// Calculate forward FFT real -> complex
226//-----------------------------------------------------------------------------
227
235
236void FFT_Implementation::calculateFFT (const FFTRealVector &in, FFTComplexVector &out)
237{
238 // resize output vector
239 if (out.size() != in.size()/2+1) out.resize(in.size()/2+1);
240
241 // check for persistent data
242 if (in.size() == 0) {
243 LogD("Calling FFT with empty vector. Skipping computation");
244 return;
245 }
246
247 // Perform the computation
248 updatePlan(in,FFTW_ESTIMATE);
249 EptAssert (in.size()==mNRC and out.size()==mNRC/2+1,"Vector consistency");
250 try {
251 std::memcpy(mRvec1,in.data(),mNRC*sizeof(double));
252 fftw_execute(mPlanRC);
253 std::memcpy(out.data(), static_cast<const void*>(mCvec2),(mNRC/2+1)*sizeof(fftw_complex));
254 }
255 catch (...) {
256 LogE("fftw_execute throwed an exception");
257 }
258}
