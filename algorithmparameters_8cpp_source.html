30void SingleAlgorithmParameters::loadDefaultParams() {
31 // load algorithm information from xml file
32 // add all params with default value if not in algorithm description
33
34 // If the algorithm is not present, we cannot load defaul parameters
35 // This usully occurs if the user loads a file with present information of an
36 // algorithm that is not implemented in the current distribution.
37 // E.g. Shipping a ept file with a newly implemented algorithm, that the user
38 // has not installed on his system.
39 if (!CalculationManager::getSingleton().hasAlgorithm(getAlgorithmName())) {
40 return;
41 }
42
43 auto info = CalculationManager::getSingleton().loadAlgorithmInformation(getAlgorithmName());
44
45 for (const AlgorithmParameterDescription &param : info->getParameters()) {
46 switch (param.getType()) {
47 case AlgorithmParameterDescription::TYPE_DOUBLE:
48 if (hasDoubleParameter(param.getID())) {continue;}
49 setDoubleParameter(param.getID(), param.getDoubleDefaultValue());
50 break;
51 case AlgorithmParameterDescription::TYPE_INT:
52 if (hasIntParameter(param.getID())) {continue;}
53 setIntParameter(param.getID(), param.getIntDefaultValue());
54 break;
55 case AlgorithmParameterDescription::TYPE_LIST:
56 if (hasStringParameter(param.getID())) {continue;}
57 setStringParameter(param.getID(), param.getStringDefaultValue());
58 break;
59 case AlgorithmParameterDescription::TYPE_BOOL:
60 if (hasBoolParameter(param.getID())) {continue;}
61 setBoolParameter(param.getID(), param.getBoolDefaultValue());
62 break;
63 default:
64 EPT_EXCEPT(EptException::ERR_NOT_IMPLEMENTED, "Parsing of default parameter not implemented");
65 break;
66 }
67 }
68}
