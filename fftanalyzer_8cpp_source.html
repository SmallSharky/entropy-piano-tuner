61
62std::pair<FFTAnalyzerErrorTypes, std::shared_ptr<Key> > FFTAnalyzer::analyse (
63 const Piano *piano,
64 FFTDataPointer finalFFT,
65 int finalKey)
66{
67 // consisty check
68 EptAssert(piano, "Piano has to be set");
69 EptAssert(finalFFT, "FFT has to exist");
70 EptAssert(finalKey >= 0, "The final key has to be set.");
71
72 std::shared_ptr<Key> key; // the key output
73
74 // dont handle an invalid fft's
75 if (!finalFFT->isValid()) {
76 LogD("Received invalid FFT to analyse. Cancel the analysis.");
77 return std::make_pair(FFTAnalyzerErrorTypes::ERR_DATA, key);
78 }
79
80
81 // Start with the analysis
82 LogV("FFTAnalyzer started");
83
84 // Map the final FFT to a logarithmically binned spectrum:
85 static SpectrumType spectrum(NumberOfBins);
86 constructLogBinnedSpectrum(finalFFT, spectrum);
87 Write("4-final-logspec.dat", spectrum);
88
89 // In the first two octaves the ground frequency f1 is very weak
90 // or even non-existing. Here we use instead the 2nd or 4th partial
91 // for the frequency estimation. This function computes how many
92 // octaves we intend to shift:
93
94 auto octaves = [] (int keynumber, int keyNumberOfA)
95 {
96 int distance = keyNumberOfA - keynumber;
97 if (distance > 36) return 2;
98 else if (distance > 24) return 1;
99 else return 0;
100 };
101
102 // First frequency estimate without inharmonicity
103 double factor = pow(2.0,octaves(finalKey, piano->getKeyboard().getKeyNumberOfA4()));
104 double f1 = factor * estimateFrequency(finalKey, piano->getConcertPitch(), piano->getKeyboard().getKeyNumberOfA4());
105 int m = Key::FrequencyToIndex(f1);
106 double f = Key::IndexToFrequency(locatePeak(spectrum,m,40)) / factor;
107
108 LogV("FFTAnalyzer: Estimated frequency f = %f, key = %d", f, finalKey);
109
110
111 if (f>20 and f<6000)
112 {
113 // Inharmonicity estimation.
114 double B = estimateInharmonicity(finalFFT, spectrum,f);
115 LogV("FFTAnalyzer: Inharmonicity B = %f", B);
116
117 // Correct frequency, taking inharmonicty into account
118 f = findAccuratePeakFrequency(finalFFT, factor*f)/factor/sqrt((1+B*factor*factor)/(1+B));
119 LogV("FFTAnalyzer: Accurate frequency f = %f", f);
120
121 // Quality check by collapsing higher partials (except of highest ocatave)
122 double Q=0;
123 if (f<2200)
124 {
125 Q = estimateQuality();
126 LogV("FFTAnalyzer: Quality measure (cents) Q = %f", Q);
127
128 double cents = estimateFrequencyShift();
129 LogV("FFTAnalyzer: frequency correction cents C = %f", cents);
130 }
131
132 // Create a new key:
133 // This is done with a shared pointer, meaning that the instance
134 // will be deleted automatically if it is no longer used
135 key.reset(new Key());
136
137 key->setRecordedFrequency(f);
138 key->setMeasuredInharmonicity(B);
139 key->setRecognitionQuality(Q);
140 key->setSpectrum(spectrum);
141 key->setRecorded(true);
142
143 auto peaks = identifyPeaks(finalFFT, spectrum,f,B);
144 // %zu is c++ standard, but windows doenst support this... workaround use int
145 LogV("FFTAnalyzer: found %i peaks.", static_cast<int>(peaks.size()));
146 key->setPeaks(peaks);
147 }
148 else
149 {
150 LogW("Frequence %f is out of bounds", f);
151 return std::make_pair(FFTAnalyzerErrorTypes::ERR_FREQUENCY_OUT_OF_BOUNDS, key);
152 }
153
154 LogV("leaving FFTAnalyzer thread.");
155
156 return std::make_pair(FFTAnalyzerErrorTypes::ERR_NONE, key);
157}

160// Determine the frequency of a known key
161//-----------------------------------------------------------------------------
162
171
172FrequencyDetectionResult FFTAnalyzer::detectFrequencyOfKnownKey (
173 FFTDataPointer finalFFT,
174 const Piano *piano,
175 const Key &key,
176 int keyIndex)
177{
178 // consisty check
179 EptAssert(piano, "Piano has to be set");
180 EptAssert(finalFFT, "FFT has to exist");
181 EptAssert(finalFFT->isValid(), "The FFT data is not valid");
182 EptAssert(keyIndex >= 0, "The final key has to be set.");
183
184 // Create a shared pointer in which the result will be stored:
185 FrequencyDetectionResult result = std::make_shared<FrequencyDetectionResultStruct>();
186
187 // This is the frequency to which we would like to tune the string
188 double targetFrequency = piano->getConcertPitch()/440.0 *
189 key.getComputedFrequency();
190
191 if (targetFrequency <= 20 or targetFrequency > 10000)
192 {
193 result->error = FFTAnalyzerErrorTypes::ERR_NO_COMPUTED_FREQUENCY;
194 return result;
195 }
196
197 // Define the frequency which corresponds to the middle of the array
198 double centerFrequency = key.getRecordedFrequency();
199 if (centerFrequency<=10) centerFrequency = targetFrequency;
200
201 // Map the final FFT to a logarithmically binned spectrum:
202 SpectrumType spectrum(NumberOfBins);
203 constructLogBinnedSpectrum(finalFFT, spectrum);
204
205 // Define the search size around the center frequency in cents
206 const int searchSize = 200;
207 TuningDeviationCurveType out(searchSize), zoomedPeak (searchSize);
208
209
210 // First method: Simply magnify the lowest peak
211 if (not key.isRecorded())
212 {
213 double maximum = 0;
214 double middle = centerFrequency;
215 if (piano->getKeyboard().getKeyNumberOfA4()-keyIndex > 24)
216 {
217 double B = piano->getExpectedInharmonicity(centerFrequency);
218 middle *= 2 * sqrt((1+4*B)/(1+B));
219 }
220
221 for (int i=0; i<searchSize; ++i)
222 {
223 int m = Key::FrequencyToRealIndex(middle) - searchSize/2 + i;
224 if (m>0 and m<static_cast<int>(spectrum.size()))
225 {
226 double value = spectrum[m] * spectrum[m];
227 if (value > maximum) maximum = value;
228 zoomedPeak[i] = value;
229 }
230 }
231 if (maximum < 1E-15)
232 {
233 result->error = FFTAnalyzerErrorTypes::ERR_NO_PEAK_AMPLITUDE;
234 return result;
235 }
236 for (auto &element : zoomedPeak) element /= maximum;
237 out = std::move(zoomedPeak);
238 }
239
240 // else if the key has been recorded explicitely we use the method developed
241 // by Christoph. It uses an inverse kernel convolution
242 else
243 {
244
245
246 // create the kernel of the key, if the key changed
247 if (&key != mCurrentKernelKey)
248 {
249 mCurrentKernel = constructKernel(key.getSpectrum());
250 mCurrentKernelKey = &key;
251 }
252
253 // compute the deviation
254 out = computeTuningDeviation(mCurrentKernel, spectrum, searchSize);
255 }
256
257 int maxIndex = MathTools::findMaximum(out);
258 double index = MathTools::weightedArithmetricMean(out, std::max(maxIndex - 10, 0), maxIndex + 10);
259 index -= searchSize / 2;
260
261 double detectedFrequency = centerFrequency * std::pow(2.0, (index) / 1200.0);
262 EptAssert(detectedFrequency > 1 && detectedFrequency < 20000, "Unallowed frequency range");
263
264 int computedIndex = MathTools::roundToInteger(log(targetFrequency / centerFrequency) * 1200 / MathTools::LOG2);
265
266
267 result->deviationInCents = static_cast<int>(index - computedIndex);
268 result->detectedFrequency = detectedFrequency;
269 result->positionOfMaximum = index;
270 result->tuningDeviationCurve = std::move(out);
271
272 LogV("Deviation %d, comp index %d", result->deviationInCents, computedIndex);
273
274 return result;
275}

278// Construct logarithmically binned spectrum
279//-----------------------------------------------------------------------------
280
295
296void FFTAnalyzer::constructLogBinnedSpectrum(FFTDataPointer fftData, SpectrumType &spectrum)
297{
298 const double b = 2.0 * fftData->fft.size() / fftData->samplingRate;
299 std::function<double(double)> mtoq = [b] (double m)
300 { return b * Key::IndexToFrequency(m); };
301 MathTools::coarseGrainSpectrum (fftData->fft,spectrum,mtoq,0.25);
302 MathTools::normalize(spectrum);
303}
304
305FFTAnalyzer::SpectrumType FFTAnalyzer::constructKernel(const SpectrumType &originalSpectrum)
306{
307 SpectrumType kernel(NumberOfBins);
308 FFTComplexVector fftOfOriginal;
309 mFFT.calculateFFT(originalSpectrum, fftOfOriginal);
310 for (FFTComplexType &c : fftOfOriginal) {
311 c = c / (c.real() * c.real() + c.imag() * c.imag());
312 }
313 mFFT.calculateFFT(fftOfOriginal, kernel);
314 return kernel;
315}
316
317TuningDeviationCurveType FFTAnalyzer::computeTuningDeviation(
318 const SpectrumType &kernel, const SpectrumType &signal, int searchSize)
319{
320 const int searchOffset = searchSize / 2;
321 TuningDeviationCurveType out(searchSize);
322
323 // in a range of 50 ct find the maximum when folding the kernel with the spectrum
324 for (int j = -searchOffset; j < searchSize - searchOffset; j++) {
325 out[j + searchOffset] = 0;
326 for (int i = 0; i < NumberOfBins; i++) {
327 out[j + searchOffset] += kernel[(i - j + NumberOfBins) % NumberOfBins] * signal[i];
328 }
329 }
330
331 return out;
332}

355//-----------------------------------------------------------------------------
356// Interpolate the position of the maximum
357//-----------------------------------------------------------------------------
358
367
368double FFTAnalyzer::interpolatePeakPosition (const SpectrumType &spectrum, int m, int width)
369{
370 EptAssert(spectrum.size()==static_cast<size_t>(NumberOfBins)
371 and spectrum.size()>0, "Inconsistent arguments");
372 if (m<width+1 or m>NumberOfBins-width-1) return 0;
373 int mmax = MathTools::findMaximum (spectrum,m-width,m+width);
374 double y1=spectrum[mmax-1], y2=spectrum[mmax], y3=spectrum[mmax+1];
375 double N = y1 - 2*y2 + y3;
376 if (N == 0) return mmax;
377 else
378 {
379 double correction = (y1-y3)/2/N;
380 if (fabs(correction)<1) return mmax+correction;
381 else return mmax;
382 }
383}

386//-----------------------------------------------------------------------------
387// Find the nearest key to a given frequency, assuming average stretch
388//-----------------------------------------------------------------------------
389
398
399int FFTAnalyzer::findNearestKey (double f, double conertPitch, int numberOfKeys, int keyNumberOfA)
400{
401 EptAssert(conertPitch>390 or conertPitch<500,"Concert pitch unreasonable.");
402 // Approximate distance in keys from A-440:
403 double d = 17.3123*log(f / conertPitch);
404 // Average stretch polynomial, giving the expected deviation in cents
405 double c =0.000019394+0.079694594*d-0.003718646*d*d+0.000450934*d*d*d + 0.000003724*d*d*d*d;
406 int k=-1; k = static_cast<int>(keyNumberOfA + d-c/100+0.5);
407 return (k>=0 and k<numberOfKeys ? k : -1);
408}

411//-----------------------------------------------------------------------------
412// Roughly frequency estimate for a given key
413//-----------------------------------------------------------------------------
414
424
425double FFTAnalyzer::estimateFrequency (int keynumber, double concertPitch, int keyNumberOfA)
426{
427 EptAssert(concertPitch>390 or concertPitch<500,"Concert pitch unreasonable.");
428 // Distance in keys from A-440:
429 double d = keynumber - keyNumberOfA;
430 // Average stretch polynomial, giving the expected deviation in cents
431 double c =0.000019394+0.079694594*d-0.003718646*d*d+
432 0.000450934*d*d*d + 0.000003724*d*d*d*d;
433 return pow(2.0, d/12+c/1200) * concertPitch;
434}

437//-----------------------------------------------------------------------------
438// Find accurate frequency of the peak in original spectrum
439//-----------------------------------------------------------------------------
440
453
454double FFTAnalyzer::findAccuratePeakFrequency(FFTDataPointer fftData, double f, int cents)
455{
456 // Define the bounds in which the peak will be searched
457 const double factor = 1.0 + 0.000577623 * cents;
458 const double b = 2.0 * fftData->fft.size() / fftData->samplingRate;
459 int q1 = MathTools::roundToInteger(b*f/factor);
460 int q2 = MathTools::roundToInteger(b*f*factor);
461
462 // search for the maximum
463 if (q1 > 0 and q2 < static_cast<int>(fftData->fft.size()))
464 {
465 double fftmax=0;
466 int qmax=q1;
467 for (int q=q1; q<q2; ++q) if (fftData->fft[q]>fftmax)
468 { fftmax = fftData->fft[q]; qmax=q; }
469 return qmax/b;
470 }
471 else return f;
472}

495//-----------------------------------------------------------------------------
496// Estimate inharmonicity
497//-----------------------------------------------------------------------------
498
514
515double FFTAnalyzer::estimateInharmonicity (FFTDataPointer fftData, SpectrumType &spectrum, double f)
516{
517 // do not evaluate inharmonicity if parameters are invalid
518 if (spectrum.size()==0 or f<20) return 0;
519
520 // do not evaluate inharmonicity in the very high treble (top octave)
521 // because this is usually very unreliable
522 if (f > 2250) return 0;
523
524 // for frequencies above 1kHz only f1 and f2 are taken into account
525 // and the inharmonicity B is calculated directly from their ratio.
526 if (f>1000)
527 {
528 double f2 = findAccuratePeakFrequency (fftData, 2.0174*f, 15);
529 double z = f2*f2/f/f;
530 if (z>4.4 or z<4) return 0;
531 double B = (4-z)/(z-16);
532 LogV("FFTAnalyzer: treble: B = %f", B);
533 return B;
534 }
535
536 // In the frequency range from 26 Hz to 1kHz the inharmonicity is computed
537 // by superposition of N higher partials and searching for the lowest
538 // Renyi entropy of the superposition.
539
540 // Define the number of partials taken into accout.
541 int N=MathTools::roundToInteger(4*(8-log(f)));
542
543 double B = 0; // Inharmonicity to be determined
544 double Hmin = 1E100; // Initial Renyi entropy very high
545 const int R = 80; // width of the observation window in cents
546
547 // Calling this function gives a first rough estimate of the inharmonicty
548 double expected_B = getExpectedInharmonicity(f);
549 LogV("FFTAnalyzer: expected B = %f", expected_B);
550
551 mOptimalSuperposition.clear();
552 for (double scan_B = expected_B/5; scan_B <= expected_B*5; scan_B*=1.03)
553 {
554 SpectrumType superposition(R,0);
555 for (int n=1; n<=N; ++n)
556 {
557 double fn = n*f*sqrt((1+scan_B*n*n)/(1+scan_B));
558 double mn = Key::FrequencyToRealIndex(fn);
559 SpectrumType partialspectrum(R,0);
560 if (mn-R/2>0 and mn+R/2<NumberOfBins)
561 {
562 for (int r=0; r<R; r++)
563 {
564 int m = static_cast<int>(mn+r-R/2);
565 EptAssert (m>=0 and m<NumberOfBins,"m invalid");
566 partialspectrum[r]=pow(spectrum[m],2);
567 }
568 MathTools::normalize(partialspectrum);
569 for (int r=0; r<R; r++) superposition[r]+=partialspectrum[r];
570 }
571 }
572
573 MathTools::normalize(superposition);
574 double H = MathTools::computeRenyiEntropy(superposition,0.1);
575 if (fabs(H)<Hmin)
576 {
577 Hmin=H;
578 B=scan_B;
579 mOptimalSuperposition = superposition;
580 Write("7-find-inharmonicity.dat",superposition);
581 }
582 }
583 LogV("FFTAnalyzer: finished estimating inharmonicity: B = %f", B);
584
585#if CONFIG_ENABLE_XMGRACE
586 //system("killall -9 xmgrace; xmgrace -maxpath 200000 -fixed 570 440 -geometry 800x660 7-find-inharmonicity.dat &");
587#endif // CONFIG_ENABLE_XMGRACE
588
589 return B;
590}

593//-----------------------------------------------------------------------------
594// estimate quality of the recorded sound
595//-----------------------------------------------------------------------------
596
608
609double FFTAnalyzer::estimateQuality ()
610{
611 // if there is no such data return quality zero
612 if (mOptimalSuperposition.size()==0) return 0;
613
614 // cut the superposition at the edges
615 const int cut = static_cast<int>(mOptimalSuperposition.size()/2)-10;
616 SpectrumType vec = SpectrumType(mOptimalSuperposition.begin()+cut,
617 mOptimalSuperposition.end()-cut);
618
619 // Compute the zeroth, first, and second moment of the distribution
620 double M0 = MathTools::computeNorm(vec);
621 if (M0==0) return 0; // if not normalizable return zero quality
622 MathTools::normalize(vec);
623 double M1 = MathTools::computeMoment(vec,1);
624 double M2 = MathTools::computeMoment(vec,2);
625
626 // Determine the variance from the moments
627 double variance = M2-M1*M1;
628 // translate heuristicaly in a measure of quality to be displayed.
629 // M0 is the fraction of the data that was not cut away.
630 // Therefore, the result is expected to be between 0 and 1
631 return M0/(1+0.1*pow(variance,1.5));
632}

635//-----------------------------------------------------------------------------
636// estimate frequency shift of the recorded sound
637//-----------------------------------------------------------------------------
638
648
649double FFTAnalyzer::estimateFrequencyShift()
650{
651 // cut out a section of 20 cents
652 size_t start = mOptimalSuperposition.size()/2-10;
653 size_t stop = mOptimalSuperposition.size()/2+10;
654 if (start>=stop or stop>=mOptimalSuperposition.size()) return 0;
655 SpectrumType vec = SpectrumType(mOptimalSuperposition.begin()+start,
656 mOptimalSuperposition.begin()+stop);
657
658 // The deviation is the first moment of the distribution in this section
659 return MathTools::computeMoment(vec,1)-10;
660}

663//-----------------------------------------------------------------------------
664// Identify peaks of the spectrum for given inharmonicity
665//-----------------------------------------------------------------------------
666
675
676FFTAnalyzer::PeakListType FFTAnalyzer::identifyPeaks (FFTDataPointer fftData,
677 const SpectrumType &spectrum,
678 const double f, const double B)
679{
680 // Define the number of peaks to be analyzed depending on the frequency
681 const int MaxNumberOfPeaks = 50;
682 int N = std::min(MaxNumberOfPeaks, static_cast<int>(10000.0/f)); // number of peaks
683
684 // Define the usual inhamonicity formula
685 auto InharmonicPartial = [] (double f, int n, double B) { return f*n*sqrt((1+B*n*n)/(1+B)); };
686
687 PeakListType peaks;
688 for (int n=1; n<=N; ++n)
689 {
690 // Compute the frequency where the peak should be
691 double fn = InharmonicPartial(f,n,B);
692
693 // Locate it in the logarithmically organized spectrum in a window of +/- 20 cents
694 int m = locatePeak (spectrum, Key::FrequencyToIndex(fn), 20);
695
696 // If succesful refine the search by looking for the maximum in the original FFT
697 if (m>0)
698 {
699 double f = Key::IndexToFrequency(m);
700 double fc = findAccuratePeakFrequency(fftData, f);
701 // Append the refined value to the list of peaks
702 peaks[fc]=spectrum[m];
703 }
704 }
705 return peaks;
706}
