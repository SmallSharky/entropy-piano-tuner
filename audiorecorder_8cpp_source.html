90int64_t AudioRecorder::write(const char *data, int64_t max_bytes) {
91 // convert data from DataType to PCMDataType
92 PacketType packet(max_bytes / sizeof(DataType));
93 const DataType *d = reinterpret_cast<const DataType *>(data);
94 PCMDataType *p = packet.data();
95 const PCMDataType *const pend = packet.data() + packet.size();
96
97 while (p != pend) {
98 *p = *d;
99 *p /= std::numeric_limits<DataType>::max();
100 ++p;
101 ++d;
102 }
103
104 pushRawData(packet);
105
106 return packet.size() * sizeof(DataType);
107}

197//-----------------------------------------------------------------------------
198// Push raw PCM data from the implementation
199//-----------------------------------------------------------------------------
200
214
215void AudioRecorder::pushRawData(const PacketType &data)
216{
217 if (data.size()==0) return;
218
219 // Forward packet to the stroboscope
220 mStroboscope.pushRawData (data);
221
222 std::lock_guard<std::mutex> lock(mCurrentPacketMutex);
223
224 // loop over all PCM values of the incoming data vector
225 for (auto &pcmvalue : data)
226 {
227 mCurrentPacket.push_back(pcmvalue);
228 mPacketM1 += pcmvalue;
229 mPacketM2 += pcmvalue*pcmvalue;
230
231 if (++mCounter > mCounterThreshold) // if packet is complete
232 {
233 // Compute the intensity (variance) of the packet and reset counter
234 double intensity = (mPacketM2-mPacketM1*mPacketM1/mCounter)/mCounter;
235 mPacketM1 = mPacketM2 = mCounter = 0;
236
237 // Convert intensity to the corresponding VU level
238 double level = MathTools::restrictToInterval(convertIntensityToLevel(intensity),0,1);
239
240 // Compute a sliding level with fast attack and slow deay
241 if (level > mSlidingLevel)
242 mSlidingLevel += (level-mSlidingLevel)*ATTACKRATE;
243 else mSlidingLevel -= (mSlidingLevel-level)*DECAYRATE;
244
245 // If muted then the shown level is zero
246 double shownLevel = (mMuted ? 0 : mSlidingLevel);
247
248 // Send shown (muted) level to the GUI VU meter
249 MessageHandler::send<MessageRecorderEnergyChanged>(MessageRecorderEnergyChanged::LevelType::LEVEL_INPUT, shownLevel);
250
251 // Switch recording process on and off according to the shown (muted) level
252 controlRecordingState (shownLevel);
253
254 // Control the input level shown at the VU meter automaticall
255 if (not mMuted) automaticControl (intensity,level);
256 }
257 }
258}

261//-----------------------------------------------------------------------------
262// Automatic noise estimation and threshold control
263//-----------------------------------------------------------------------------
264
277
278void AudioRecorder::automaticControl (double intensity, double level)
279{
280 if (intensity == 0) return;
281
282 // Limit the maximal raw pcm intensity.
283 // A sine wave with maximal amplitude +/- 1 has the integrated intensity 1/2.
284 // Therefore, we reduce the hardware input level whenever this value
285 // is surpassed. This down-regulation is irreversible.
286 if (intensity > 0.45) mAudioInterface->setGain(mAudioInterface->getGain() * 0.9);
287
288 // The level (passed as parameter) is a power law funtion of the intensity
289 // scaled with the parameter mGain. The parameter mGain controls the overall
290 // input amplification. It can move in both directions.
291
292 // if level exceeds predefined cutoff turn gain down
293 if (level > LEVEL_CUTOFF) mGain *= 0.9;
294
295 // compute intensity histogram in decibel
296 double dB = 4.34294 * log(intensity);
297 mIntensityHistogram[MathTools::roundToInteger(dB)]++;
298
299 // after 10 packets if the histogram has enough entries
300 if ((++mPacketCounter)%10==0 and mIntensityHistogram.size()>5)
301 {
302 // determine left and right boundary in a soft way
303 double norm=0, MIN=0, MAX=0;
304 for (auto &e : mIntensityHistogram)
305 {
306 norm += e.second;
307 MIN += pow(fabs(static_cast<double>(e.first)),10) * e.second;
308 MAX += exp(e.first) * e.second;
309 }
310 if (norm==0) return;
311 double dBoff = -pow(MIN/norm,1.0/10) + DB_OFF;
312 double dBmax = log(MAX/norm);
313
314 // slowly adjust the gain according to the upper edge
315 double gain_target = exp(-dBmax/4.34294/2.0);
316 mGain += 0.02*(gain_target-mGain);
317
318 // slowly adjust the noise cutoff according to the lower edge
319 double newStopLevel = convertIntensityToLevel(exp (dBoff/4.34294));
320 if (newStopLevel < LEVEL_RETRIGGER and fabs(newStopLevel-mStopLevel)>0.01)
321 {
322 // notify that stop level changed (reposition off level in mainwindow)
323 mStopLevel += 0.1*(newStopLevel-mStopLevel);
324 MessageHandler::send<MessageRecorderEnergyChanged>(
325 MessageRecorderEnergyChanged::LevelType::LEVEL_OFF, mStopLevel);
326 }
327
328 // After 10 packages decrease the whole histogram by a factor.
329 // This ensures that new data gradually overwrites older data.
330 if (mPacketCounter >= 10) for (auto &e: mIntensityHistogram) e.second /= 1.018;
331 }
332}

351
352void AudioRecorder::controlRecordingState (double level)
353{
354 // check for recording end
355 // ------------------------------------------------------------------------
356
357 if (mRecording and level < mStopLevel)
358 {
359 mRecording = false;
360 mRestartable = true;
361 MessageHandler::send(Message::MSG_RECORDING_ENDED);
362 LogI("Recording stopped");
363 }
364
365 // check for recording start (only if not in standby mode)
366 // ------------------------------------------------------------------------
367
368 if (mStandby or mWaiting) return;
369
370 // If the level falls below LEVEL_RETRIGGER, the status of
371 // the adapter is declared as restartable:
372 if (not mRestartable and mSlidingLevel < LEVEL_RETRIGGER) mRestartable = true;
373
374 // If the adapter is restartable or crosses the trigger recording starts
375 if (level>LEVEL_TRIGGER and mRestartable)
376 {
377 if (mRecording) LogI("Recording retriggered");
378 else LogI("Recording started");
379 mRecording = true;
380 mRestartable = false;
381 mWaiting = true;
382 MessageHandler::send(Message::MSG_RECORDING_STARTED);
383 }
384}

387//-----------------------------------------------------------------------------
388// Cut away silence at the beginning of the buffer
389//-----------------------------------------------------------------------------
390
399
400void AudioRecorder::cutSilence (PacketType &packet)
401{
402 // determine the maximum amplitude in the packet
403 double maxamplitude = 0;
404 for (auto &y : packet) if (fabs(y)>maxamplitude) maxamplitude = fabs(y);
405 double trigger = std::min(0.2,maxamplitude*maxamplitude/100);
406
407 int w = getSampleRate() / 40; // section width of 0.025 sec
408 int sections = static_cast<int>(packet.size()) / w; // number of sections
409 if (sections < 2) return; // required: at least two sections
410 size_t entries_to_delete = 0; // number of sections to be deleted
411 for (int sec = 0; sec < sections; ++sec) // scan sectors
412 {
413 double energy = 0;
414 for (int i = sec*w; i < (sec+1)*w; ++i) energy += packet[i] * packet[i];
415 if (energy / w < trigger) entries_to_delete += w;
416 else break;
417 }
418 // Remove the first silent sectors:
419 EptAssert(entries_to_delete <= packet.size(),"inconsistent numbers of elements");
420 if (entries_to_delete > 0)
421 packet.erase(packet.begin(), packet.begin() + entries_to_delete);
422}
