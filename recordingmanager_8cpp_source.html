50//-----------------------------------------------------------------------------
51// Message handler (for standby and stroboscope)
52//-----------------------------------------------------------------------------
53
58
59void RecordingManager::handleMessage(MessagePtr m)
60{
61 switch (m->getType())
62 {
63 case Message::MSG_PROJECT_FILE:
64 {
65 // In case that e.g. a new file was opened:
66 auto mpf(std::static_pointer_cast<MessageProjectFile>(m));
67 mPiano = &mpf->getPiano();
68 mKeyNumberOfA4 = mPiano->getKeyboard().getKeyNumberOfA4();
69 mSelectedKey = nullptr;
70 mNumberOfSelectedKey = -1;
71 updateStroboscopicFrequencies();
72 break;
73 }
74 case Message::MSG_SIGNAL_ANALYSIS:
75 {
76 auto msa(std::static_pointer_cast<MessageSignalAnalysis>(m));
77 if (msa->status() == MessageSignalAnalysis::Status::ENDED) {
78 // Tell the audio recorder to wait after completed analysis
79 // in order to avoid self-recording of the echo sound
80 mAudioRecorder->setWaitingFlag (false);
81 }
82 break;
83 }
84 case Message::MSG_MODE_CHANGED:
85 {
86 // Change of the operation mode:
87 auto mmc(std::static_pointer_cast<MessageModeChanged>(m));
88 mOperationMode = mmc->getMode();
89 switch (mOperationMode)
90 {
91 case MODE_TUNING:
92 mStroboscope->start();
93 mAudioRecorder->setStandby(false);
94 break;
95 case MODE_RECORDING:
96 mStroboscope->stop();
97 mAudioRecorder->setStandby(false);
98 break;
99 default:
100 mStroboscope->stop();
101 mAudioRecorder->setStandby(true);
102 break;
103 }
104 break;
105 }
106 case Message::MSG_KEY_SELECTION_CHANGED:
107 {
108 // If a key is selected update the stroboscopic frequencies
109 if (mOperationMode == MODE_TUNING)
110 {
111 auto message(std::static_pointer_cast<MessageKeySelectionChanged>(m));
112 mSelectedKey = message->getKey();
113 mNumberOfSelectedKey = message->getKeyNumber();
114 updateStroboscopicFrequencies();
115 }
116 break;
117 }
118 case Message::MSG_RECORDING_STARTED:
119 {
120 if (mOperationMode == MODE_TUNING)
121 mAudioRecorder->getStroboscope()->setFramesPerSecond(FPS_FAST);
122 break;
123 }
124 case Message::MSG_RECORDING_ENDED:
125 {
126 if (mOperationMode == MODE_TUNING)
127 mAudioRecorder->getStroboscope()->setFramesPerSecond(FPS_SLOW);
128 break;
129 }
130 default:
131 break;
132 }
133}

136//-----------------------------------------------------------------------------
137// Update the stroboscopic frequencies
138//-----------------------------------------------------------------------------
139
147
148void RecordingManager::updateStroboscopicFrequencies()
149{
150 std::vector<double> ftab;
151 if (mSelectedKey)
152 {
153 const double fc = mSelectedKey->getComputedFrequency();
154 const double fr = mSelectedKey->getRecordedFrequency();
155 const double cp = mPiano->getConcertPitch();
156 if (fc > 0 and fr > 0)
157 {
158 const Key::PeakListType peaks = mSelectedKey->getPeaks();
159 // This is the formula determining the number of partials shown in the stroboscope:
160 const int numberOfStroboscopicPartials = std::max(1,1+(mKeyNumberOfA4+6+24-mNumberOfSelectedKey)/6);
161
162 // if peaks are available
163 if (peaks.size()>0)
164 {
165 //const double f1 = peaks.begin()->first;
166 int N = 0;
167
168 for (auto &e : peaks)
169 {
170 if (++N > numberOfStroboscopicPartials) break;
171 // Push partials adjusted by concert pitch
172 ftab.push_back(e.first*fc/fr*cp/440.0);
173 }
174 }
175 else // if no peaks are available
176 {
177 // Push partials with the expected harmonic spectrum
178 const double B = mPiano->getExpectedInharmonicity (fc);
179 for (int n=1; n<=3; ++n) ftab.push_back(n*fc*sqrt((1+B*n*n)/(1+B)));
180 }
181 }
182 }
183 mAudioRecorder->getStroboscope()->setFrequencies(ftab);
184}
