33// This is the actual worker function in which you have to define your algorithm.
34// It will be started in a separate std::thread.
35// It might happen that the user wants to cancel the algorithm even though it is not
36// finished yet. Therefore you have to set exit marks using CHECK_CANCEL_THREAD, at
37// whom the algorithm can exit and the thread will be stopped.
38void ExampleAlgorithm::algorithmWorkerFunction()
39{
40 // use check cancel thread flags to mark exit points (if the user wants to cancel)
41 CHECK_CANCEL_THREAD;
42
43 // do the calculation here
44
45 // for example we just to a equidistant tuning
46 for (int i = 0; i < mNumberOfKeys; ++i)
47 {
48 // usually the exit mark should be set at each time before you make some kind of longer calculation,
49 // that can't be interrupted.
50 CHECK_CANCEL_THREAD;
51
52 // show the calculation progress to the user
53 showCalculationProgress(static_cast<double>(i) / mNumberOfKeys);
54
55 // sleep for 0.2 sec so that you can see the changes in the tuning curve and test to cancel the algorithm
56 msleep(200);
57
58 // set the tuning curve
59 updateTuningCurve(i, mPiano.getEqualTempFrequency(i, 0, mConcertPitchParam));
60 }
61}
