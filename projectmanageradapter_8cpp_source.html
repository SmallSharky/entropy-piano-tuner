113// Menu command: New file
114//-----------------------------------------------------------------------------
115
122
123ProjectManagerAdapter::Results ProjectManagerAdapter::onNewFile()
124{
125
126 if (checkForNoChanges() == R_CANCELED) {return R_CANCELED;}
127
128 // create new file
129 mCurrentFilePath.clear();
130 mCore->getPianoManager()->getPiano() = Piano();
131 fillNew(mCore->getPianoManager()->getPiano());
132
133 setChangesInFile(true);
134 MessageHandler::send<MessageProjectFile>(MessageProjectFile::FILE_CREATED,
135 mCore->getPianoManager()->getPiano());
136
137 // edit this new file
138 if (editFile() == R_ACCEPTED)
139 {
140 MessageHandler::send<MessageProjectFile>(MessageProjectFile::FILE_EDITED,
141 mCore->getPianoManager()->getPiano());
142 }
143 return R_ACCEPTED;
144}

235
236bool ProjectManagerAdapter::onQuit()
237{
238 if (mChangesInFile)
239 {
240 switch (askForSaving())
241 {
242 case R_NO:
243 // dont save file, but resume
244 break;
245 case R_CANCELED:
246 return false;
247 default:
248 // save the file
249 if (onSaveFile() == R_CANCELED) {
250 // cancel aswell
251 return R_CANCELED;
252 }
253 break;
254 }
255 }
256 LogI("Quitting accepted");
257 return true;
258}

345ProjectManagerAdapter::Results ProjectManagerAdapter::checkForNoChanges()
346{
347 if (mChangesInFile)
348 {
349 switch (askForSaving())
350 {
351 case R_NO:
352 // dont save file, but resume
353 break;
354 case R_CANCELED:
355 return R_CANCELED;
356 default:
357 // save the file first

417//-----------------------------------------------------------------------------
418// Save file
419//-----------------------------------------------------------------------------
420
432
433ProjectManagerAdapter::Results ProjectManagerAdapter::saveFile(const FileDialogResult &fileInfo, FileSaveType fst)
434{
435 EptAssert(fileInfo.isValid(), "File type not valid.");
436
437 try // catching possible exception errors
438 {
439 // Try to save the piano
440 writePianoFile(fileInfo, mCore->getPianoManager()->getPiano());
441
442 LogI("File saved!");
443
444 if (fst == FST_SAVE) {
445 mCurrentFilePath = fileInfo.path; // remember current path
446 setChangesInFile(false); // after saving the status is that there are no changes
447
448 // Tell the other modules that the file has been saved
449 MessageHandler::send<MessageProjectFile>(MessageProjectFile::FILE_SAVED,
450 mCore->getPianoManager()->getPiano());
451 }
452 }
453 catch (const EptException &e)
454 {
455 LogW("Exception during saving a file: %s", e.what());
456 // show message box, that there is an error
457 showSaveError();
458 return R_CANCELED;
459 }
460 return R_ACCEPTED;
461}

464//-----------------------------------------------------------------------------
465// Read (open) piano file
466//-----------------------------------------------------------------------------
467
475
476ProjectManagerAdapter::Results ProjectManagerAdapter::openFile(const FileDialogResult &fileInfo, bool cached)
477{
478 try
479 {
480 readPianoFile(fileInfo, &mCore->getPianoManager()->getPiano());
481
482 LogI("File opened!");
483 if (cached)
484 {
485 // file is in cache, we need to save it at a new position
486 mCurrentFilePath.clear();
487 setChangesInFile(true);
488 }
489 else
490 {
491 mCurrentFilePath = fileInfo.path;
492 setChangesInFile(false);
493 }
494 MessageHandler::send<MessageProjectFile>(MessageProjectFile::FILE_OPENED, mCore->getPianoManager()->getPiano());
495 }
496 catch (const EptException &e)
497 {
498 LogW("Exception during opening a file: %s", e.what());
499 // show message box, that there is an error
500 showOpenError();
501 return R_CANCELED;
502 }
503 return R_ACCEPTED;
504}
