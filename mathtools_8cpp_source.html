40// Determine the raw moments of a distribution
41//-----------------------------------------------------------------------------
42
47
48double MathTools::computeMoment (const std::vector<double> &v, const int n)
49{
50 EptAssert (n>=0 and v.size()>0, "Parameters inconsistent");
51 double norm=0,sum=0;
52 for (size_t i=0; i<v.size(); ++i)
53 {
54 norm += v[i];
55 sum += v[i] * pow(1.0*i,1.0*n);
56 }
57 EptAssert (norm>0, "The norm should be positive");
58 return sum / norm;
59}

62//-----------------------------------------------------------------------------
63// Compute Shannon entropy
64//-----------------------------------------------------------------------------
65
73
74double MathTools::computeEntropy (const std::vector<double> &v)
75{
76 EptAssert(v.size()>0,"The entropy of a vector with zero length is meaningless.");
77 double sum=0;
78 for (auto x : v) sum -= (x>0 ? x*log(x) : 0);
79 return sum;
80}

83//-----------------------------------------------------------------------------
84// Compute Renyi entropy
85//-----------------------------------------------------------------------------
86
98
99double MathTools::computeRenyiEntropy (const std::vector<double> &v, const double q)
100{
101 EptAssert(v.size()>0,"The entropy of a vector with zero length is meaningless.");
102 EptAssert(q>0,"The Renyi deformation parameter should be positive.");
103 if (q==1) return computeEntropy (v);
104 double sum=0;
105 for (auto x : v) sum += pow(x,q);
106 return log(sum) / (1-q);
107}

125{
126 double norm = computeNorm(vec);
127 EptAssert (norm!=0,"Vectors with norm zero cannot be normalized");
128 for (auto &x : vec) x /= norm;

132//-----------------------------------------------------------------------------
133// Map a probability distribution from one vector to another
134//-----------------------------------------------------------------------------
135
150
151
152void MathTools::coarseGrainSpectrum (const std::vector<double> &X,
153 std::vector<double> &Y,
154 std::function<double(double y)> f,
155 double exponent)
156{
157 assert(X.size()>0 and Y.size()>0);
158 double xs1 = f(-0.5);
159 int x1 = std::max<int>(0, roundToInteger(xs1));
160 double leftarea = (x1-xs1+0.5)*X[x1];
161 for (int y=0; y<static_cast<int>(Y.size()); ++y)
162 {
163 double xs2 = f(y+0.5);
164 int x2 = std::min<int>(roundToInteger(xs2), static_cast<int>(X.size()) - 1);
165 double sum=0;
166 for (int x=x1+1; x<=x2; ++x) sum += X[x];
167 double rightarea = (x2-xs2+0.5)*X[x2];
168 Y[y] = (sum + leftarea - rightarea) * pow(xs1*xs2,exponent);
169 x1=x2; xs1=xs2; leftarea=rightarea;
170 }
171}

196double MathTools::findSmoothedMaximum (const std::vector<double> &x)
197{
198 auto maxElem = std::max_element(std::next(x.begin()), std::prev(x.end()));
199 int maxIndex = static_cast<int>(std::distance(x.begin(), maxElem));
200 // interpolate with neighbours using a parabolas maximum
201 double y1 = *std::prev(maxElem);
202 double y2 = *maxElem;
203 double y3 = *std::next(maxElem);
204 double x1 = maxIndex - 1, x2 = maxIndex, x3 = maxIndex + 1;
205
206 // compute max of parabola
207 return (x3 * x3 * (y2 - y1) + x2 * x2 * (y1 - y3) + x1 * x1 * (y3 - y2)) / 2.0 / (x3 * (y2 - y1) + x2 * (y1 - y3) + x1 * (y3 - y2));
208}

210double MathTools::weightedArithmetricMean(const std::vector<double> &Y, size_t start, size_t end)
211{
212 double norm = 0;
213 double mean = 0;
214 end = std::min(end, Y.size());
215 for (size_t i = start; i < end; ++i)
216 {
217 norm += Y[i] * Y[i];
218 mean += Y[i] * Y[i] * i;
219 }
220
221 return mean / norm;
222}
