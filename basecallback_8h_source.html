64public:
70 virtual ~BaseCallbackManager() {
71 while (mListeners.size() > 0) {
72 auto l = mListeners.front();
73 mListeners.pop_front();
74 removeListener(l);

84 void addListener(BaseCallbackInterface *listener) {
85 auto it = std::find(mListeners.begin(), mListeners.end(), listener);
86 if (it == mListeners.end()) {
87 mListeners.push_back(listener);
88 listener->addCallbackManager(this);
89 }
90 }
91
98 void removeListener(BaseCallbackInterface *listener) {
99 auto it = std::find(mListeners.begin(), mListeners.end(), listener);
100 if (it != mListeners.end()) {
101 mListeners.erase(it);
102 listener->removeCallbackManager(this);
103 }
104 }
105
110 const std::list<BaseCallbackInterface*> &listeners() const {return mListeners;}
111private:
112
114 std::list<BaseCallbackInterface*> mListeners;
115};
