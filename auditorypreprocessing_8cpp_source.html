60//-----------------------------------------------------------------------------
61// Check consistency of dataset
62//-----------------------------------------------------------------------------
63
71bool AuditoryPreprocessing::checkDataConsistency()
72{
73 // First check whether the number of keys is consistent.
74 EptAssert(mKeys.size() > 0, "Piano should have at least one key");
75 EptAssert(mKeys.size() == static_cast<size_t>(mNumberOfKeys),
76 "Key vector length mismatch");
77
78 // Next let us see whether all keys have been recorded.
79 bool allkeysrecorded = true;
80 bool somekeysrecorded = false;
81 for (Key key : mKeys)
82 {
83 if (key.isRecorded()) somekeysrecorded = true;
84 else allkeysrecorded = false;
85 }
86 if (not somekeysrecorded)
87 {
88 MessageHandler::send<MessageCaluclationProgress>
89 (MessageCaluclationProgress::CALCULATION_FAILED,
90 MessageCaluclationProgress::CALCULATION_ERROR_NO_DATA);
91 LogW("Calculation started without data");
92 return false;
93 }
94 if (not allkeysrecorded)
95 {
96 MessageHandler::send<MessageCaluclationProgress>
97 (MessageCaluclationProgress::CALCULATION_FAILED,
98 MessageCaluclationProgress::CALCULATION_ERROR_NOT_ALL_KEYS_RECORDED);
99 LogW("Not all keys have been recorded");
100 return false;
101 }
102
103 // Thirdly, let us see whether the frequencies and inharmonicities
104 // are within reasonable bounds
105 bool consistent = true;
106 for (int keynumber=0; keynumber<mNumberOfKeys; keynumber++)
107 {
108 Key &key = mKeys[keynumber];
109 double f = key.getRecordedFrequency();
110 if (f<20 or f>20000)
111 {
112 LogW("Key %d: Frequency f=%f out of range.",keynumber,f);
113 consistent = false;
114 }
115 double B = key.getMeasuredInharmonicity();
116 if (B<0 or B>1)
117 {
118 LogW("Key %d: Inharmonicity B=%f out of range.",keynumber,B);
119 consistent=false;
120 }
121 Key::SpectrumType &spectrum = key.getSpectrum();
122 if (spectrum.size() != static_cast<size_t>(Key::NumberOfBins))
123 {
124 LogW("Key %d: Logspec size is %d, expected %d.",
125 keynumber,
126 static_cast<int>(spectrum.size()),
127 static_cast<int>(Key::NumberOfBins));
128 consistent=false;
129 }
130 else if (MathTools::computeNorm(spectrum)==0)
131 {
132 LogW("Key %d: Logspec norm = %f.",keynumber,
133 MathTools::computeNorm(spectrum));
134 consistent=false;
135 }
136 }
137
138 // If data is found to be inconsistent send a corresponding message.
139 if (not consistent)
140 {
141 MessageHandler::send<MessageCaluclationProgress>
142 (MessageCaluclationProgress::CALCULATION_FAILED,
143 MessageCaluclationProgress::CALCULATION_ERROR_KEY_DATA_INCONSISTENT);
144 }
145 return consistent;
146}

167//-----------------------------------------------------------------------------
168// Compute frequency of inharmonic partial
169//-----------------------------------------------------------------------------
170
181
182double AuditoryPreprocessing::getInharmonicPartial (double n, double f1, double B)
183{
184 return f1 * n * sqrt((1+B*n*n)/(1+B));
185}

188//-----------------------------------------------------------------------------
189// Compute the inharmonic index for a given frequency
190//-----------------------------------------------------------------------------
191
202
203double AuditoryPreprocessing::getInharmonicPartialIndex (double f, double f1, double B)
204{
205 double x=f/f1;
206 if (B==0) return x;
207 else return 0.7071067811865475*sqrt((-1 + sqrt(1 + 4*B*(1+B)*x*x))/B);
208}

211//-----------------------------------------------------------------------------
212// Clean spectrum
213//-----------------------------------------------------------------------------
214
229
230void AuditoryPreprocessing::cleanSpectrum (Key &key)
231{
232 SpectrumType &spectrum = key.getSpectrum();
233 const int M = static_cast<int>(spectrum.size());
234 const double f = key.getRecordedFrequency();
235 const double B = key.getMeasuredInharmonicity();
236
237 auto wave = [f,B,this] (int m)
238 { return cos(MathTools::PI*getInharmonicPartialIndex(mtof(m),f,B)); };
239 auto envelope = [wave,f] (int m)
240 { return pow(fabs(wave(m)),200.0/pow(mtof(m)/f,1.5)); };
241
242 for (int m=0; m<M; m++) spectrum[m] *= envelope(m);
243}

260 const int lowcutindex = std::min<int>(static_cast<int>((5*ftom(f)))/6,static_cast<int>(NumberOfBins));
261 for (int m=0; m<lowcutindex; m++) spectrum[m]=0;
262}

266//-----------------------------------------------------------------------------
267// Initialize the filter function
268//-----------------------------------------------------------------------------
269
279
280void AuditoryPreprocessing::initializeSPLAFilter()
281{
282 mdBA.clear();
283 mdBA.resize(NumberOfBins);
284 for (uint m=0; m<NumberOfBins; ++m)
285 {
286 double f = mtof(m);
287 double Ra = 12200.0*12200.0*f*f*f*f / (f*f+20.6*20.6) /
288 sqrt((f*f+107.7*107.7)*(f*f+737.9*737.9)) / (f*f+12200.0*12200.0);
289 mdBA[m] = 2.0+20*log10(Ra);
290 //std::cout << f << "\t" << mdBA[m] << std::endl;
291 }
292}

295//-----------------------------------------------------------------------------
296// Compute A-weighted sound pressure level SPLA
297//-----------------------------------------------------------------------------
298
310
311void AuditoryPreprocessing::convertToSPLA (SpectrumType &spectrum)
312{
313 if (mdBA.size()==0) initializeSPLAFilter();
314 EptAssert(mdBA.size()==NumberOfBins,"mdBA should be initialized.");
315 const double I0 = 1E-7; // auditory threshold intensity
316 for (uint m=0; m<NumberOfBins; ++m)
317 {
318 double SPLA = 10 * log10(spectrum[m]/I0) + mdBA[m];
319 if (SPLA<0) spectrum[m]=0;
320 else spectrum[m] = I0 * pow(10.0, SPLA/10.0);
321 }
322}

325//-----------------------------------------------------------------------------
326// extrapolate missing inharmonicity estimates
327//-----------------------------------------------------------------------------
328
340
341void AuditoryPreprocessing::extrapolateInharmonicity()
342{
343 const int firstkey = mKeyNumberOfA4-8;
344 double K=0, Y=0, KK=0, KY=0, N=0, BE=0;
345 for (int k = firstkey; k < mNumberOfKeys; ++k)
346 {
347 if (N>1)
348 {
349 double a = (N*KY-K*Y)/(N*KK-K*K);
350 double b = (KK*Y-K*KY)/(N*KK-K*K);
351 BE = exp(a*k+b);
352 }
353 double B = mKeys[k].getMeasuredInharmonicity();
354 bool B_is_valid = (B>0);
355 if (B>0 and BE>0 and N>5) if (fabs(log(B/BE))>0.2) B_is_valid=false;
356 if (B_is_valid)
357 {
358 double y = log(B);
359 K+=k; Y+=y; KK+=k*k; KY+=k*y; N++;
360 }
361 else
362 {
363 if (BE==0)
364 {
365 double f1 = mPiano.getEqualTempFrequency(k,0,440);
366 BE = mPiano.getExpectedInharmonicity(f1);
367 }
368 mKeys[k].setMeasuredInharmonicity(BE);
369 }
370 }
371}

374//-----------------------------------------------------------------------------
375// Improve high-frequency spectral lines
376//-----------------------------------------------------------------------------
377
384
385void AuditoryPreprocessing::improveHighFrequencyPeaks ()
386{
387 int start = mKeyNumberOfA4;
388 for (int k = start; k < mNumberOfKeys; k++)
389 {
390 Key &key = mKeys[k];
391 SpectrumType &spectrum = key.getSpectrum();
392 double f = key.getRecordedFrequency();
393 double B = key.getMeasuredInharmonicity();
394 if (f<=0 or B<=0) continue;
395 int m = MathTools::roundToInteger(ftom(f));
396 if (m<0 or m>static_cast<int>(NumberOfBins)) continue;
397 double factor = (1.0*(k-start))/(mNumberOfKeys-start);
398 double intensity = spectrum[m] * factor;
399 for (int n=2; n<=6; ++n)
400 {
401 double fn = getInharmonicPartial(n,f,B);
402 if (fn<20 or fn>10000) continue;
403 int mn = MathTools::roundToInteger(ftom(fn));
404 for (int i=mn-10; i<=mn+10; ++i)
405 if (i>=0 and i < static_cast<int>(NumberOfBins))
406 spectrum[i] = intensity * pow(4,-n) * exp(-0.1*(i-mn)*(i-mn));
407 }
408 }
409}

413//-----------------------------------------------------------------------------
414// Mollify
415//-----------------------------------------------------------------------------
416
417void AuditoryPreprocessing::applyMollifier (Key &key)
418{
419 SpectrumType &spectrum = key.getSpectrum();
420 SpectrumType copy = spectrum;
421 //double f1 = key.getRecordedFrequency();
422
423
424 //SpectrumType copy = spectrum;
425 int M = static_cast<int>(NumberOfBins);
426 for (int m=0; m<M; ++m)
427 {
428 double f = mtof(m);
429 const double df=55.0/f+f/2000.0;
430 int dm = MathTools::roundToInteger(ftom(f+df)) - m;
431 double sum=0,norm=0;
432 for (int ms = std::max(1,m-3*dm); ms<=std::min(m+3*dm,M-1); ms++)
433 {
434 double weight = exp(-1.0*(ms-m)*(ms-m)/dm/dm);
435 norm+=weight;
436 sum+=copy[ms]*weight;
437 }
438 if (norm>0) spectrum[m]=sum/norm;
439 }
