102
113
114void Core::init(CoreInitialisationAdapter *initAdapter)
115{
116 EptAssert(initAdapter, "At least the default adapter has to be provided");
117
118 if (mInitialized) return; // If already initialized return
119
120 initAdapter->create(); // Open initialization message box
121
122 initAdapter->updateProgress (0); // empty
123
124 initAdapter->updateProgress (11); // Initialize recorder
125 mRecorderInterface->init();
126 mRecorderInterface->setDevice(&mAudioRecoder);
127
128 initAdapter->updateProgress (22); // Initialize player
129 mPlayerInterface->init();
130
131 initAdapter->updateProgress (33); // Start the player, so that the startup sound is played
132 mPlayerInterface->start();
133
134 initAdapter->updateProgress (44); // Initialize the signal analyzer
135 mSignalAnalyzer.init();
136
137 initAdapter->updateProgress (55); // Initialize the sound generator
138 if (mEnableSoundGenerator) {
139 mSoundGenerator.reset(new SoundGenerator(mPlayerInterface));
140 mSoundGenerator->init();
141 } else {
142 LogI("SoundGenerator is disabled. Possibly low physical memory!");
143 }
144
145 initAdapter->updateProgress (65); // Initialize the recording manager
146 mRecordingManager.init();
147
148 initAdapter->updateProgress (75); // Initialze the MIDI system
149
150 initAdapter->updateProgress (87); // Open the default MIDI port
151
152 initAdapter->updateProgress (100);
153
154 mInitialized = true; // set initialization flag and
155 initAdapter->destroy(); // remove the init message box
156}
