57//-----------------------------------------------------------------------------
58// map frequency to real-valued logspec index
59//-----------------------------------------------------------------------------
60
71
72double Key::FrequencyToRealIndex (double f)
73{
74 if (f < 0) {
75 LogD("Frequency <= 0Hz detected.");
76 return -1;
77 } else if (f > 20000) {
78 LogD("Frequency higher than 20KHz detected.");
79 }
80
81 static const double lnfmin = log(fmin);
82 return BinsPerOctave * (log(f) - lnfmin) / MathTools::LOG2;
83}
84
85//-----------------------------------------------------------------------------
86// map frequency to integer logspec index
87//-----------------------------------------------------------------------------
88
96
97int Key::FrequencyToIndex (double f)
98{ return MathTools::roundToInteger(FrequencyToRealIndex(f)); }

101//-----------------------------------------------------------------------------
102// map logspec index to frequency
103//-----------------------------------------------------------------------------
104
110
111double Key::IndexToFrequency (double m)
112{
113 // m may be negative aswell here
114 return fmin * pow(2, m / BinsPerOctave);
115}
