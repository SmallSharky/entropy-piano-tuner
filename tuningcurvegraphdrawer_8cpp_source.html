92
93void TuningCurveGraphDrawer::handleMessage(MessagePtr m)
94{
95 EptAssert(m, "Message has to exist");
96
97 switch (m->getType())
98 {
99 case Message::MSG_PROJECT_FILE:
100 {
101 // get the piano data and its most important parameters:
102 auto mpf(std::static_pointer_cast<MessageProjectFile>(m));
103 mPiano = &mpf->getPiano();
104 mConcertPitch = mPiano->getConcertPitch();
105 mNumberOfKeys = mPiano->getKeyboard().getNumberOfKeys();
106 mKeyNumberOfA4 = mPiano->getKeyboard().getKeyNumberOfA4();
107 redraw();
108 break;
109 }
110 case Message::MSG_KEY_DATA_CHANGED:
111 {
112 // upate the marker positions upon key data change
113 auto mkdc(std::static_pointer_cast<MessageKeyDataChanged>(m));
114 int index = mkdc->getIndex();
115 updateMarkerPosition(index, ROLE_COMPUTED_FREQUENCY);
116 updateMarkerPosition(index, ROLE_INHARMONICITY);
117 updateMarkerPosition(index, ROLE_RECORDED_FREQUENCY);
118 updateMarkerPosition(index, ROLE_TUNED_FREQUENCY);
119 updateMarkerPosition(index, ROLE_OVERPULL);
120 break;
121 }
122 case Message::MSG_CLEAR_RECORDING:
123 {
124 redraw(true);
125 break;
126 }
127 case Message::MSG_MODE_CHANGED:
128 {
129 auto mmc(std::static_pointer_cast<MessageModeChanged>(m));
130 mOperationMode = mmc->getMode();
131 redraw(true);
132 break;
133 }
134 default:
135 break;
136 }
137}

140//-----------------------------------------------------------------------------
141// Convert cents to Y-coordinate
142//-----------------------------------------------------------------------------
143
154
155double TuningCurveGraphDrawer::convertCentsToY (double cents)
156{
157 if (cents>rangeInCents) return 0;
158 else if (cents<-rangeInCents) return 1;
159 else if (mOperationMode == MODE_TUNING)
160 return 0.5 - 0.5 / atan(3.1) * atan(3*cents/rangeInCents);
161 else return 0.5 - 0.5*cents/rangeInCents;
162}

164//-----------------------------------------------------------------------------
165// Manually edit tuning curve by mouse click
166//-----------------------------------------------------------------------------
167
178
179void TuningCurveGraphDrawer::manuallyEditTuningCurveByClick (double relX, double relY)
180{
181 if (relX < 0 or relX > 1 or relY < 0 or relY > 1) return; // out of range
182
183 int keynumber = static_cast<int>(mNumberOfKeys * relX);
184 if (keynumber < 0 or keynumber >= mNumberOfKeys) return;
185
186 if (keynumber == mKeyNumberOfA4)
187 {
188 LogI("The pitch of A4 cannot be changed manually.");
189 return;
190 }
191 double yfrac=1 - relY;
192 int cents = MathTools::roundToInteger(rangeInCents * (2 * yfrac - 1));
193 LogW("Set tuningcurve manually keynumber=%d, cents=%d.",keynumber,cents);
194 MessageHandler::send<MessageChangeTuningCurve>
195 (keynumber,mPiano->getDefiningTempFrequency(keynumber,cents,440));
196}

199//-----------------------------------------------------------------------------
200// Draw a grid in the background
201//-----------------------------------------------------------------------------
202
210
211void TuningCurveGraphDrawer::drawGrid ()
212{
213 // Draw a horizontal background line for each 10 cents
214 for (int cents=-rangeInCents+1; cents<rangeInCents; cents++)
215 if (cents%10==0 or (abs(cents)<10 and mOperationMode==MODE_TUNING))
216 {
217 double y = convertCentsToY(cents);
218 auto type = (cents ? centlines : middleline);
219 mGraphics->drawLine(0, y, 1, y, type)->setItemRole(ROLE_GRID);
220 }
221 // Draw vertical lines according to the keys
222 if (mNumberOfKeys > 0) {
223 const double dx = 1.0 / mNumberOfKeys;
224 for (int i = 0; i <= mNumberOfKeys; i++)
225 mGraphics->drawLine(i*dx, 0,i*dx, 1, gridcolor)->setItemRole(ROLE_GRID);
226 }
227
228 // Draw green background in tuning mode
229 if (mOperationMode == MODE_TUNING) {
230 const double allowedDeviationInCents = 3;
231 double allowedStartCents = convertCentsToY(allowedDeviationInCents);
232 GraphicsItem *item = mGraphics->drawFilledRect(0, allowedStartCents, 1,
233 convertCentsToY(-allowedDeviationInCents) - allowedStartCents,
234 GraphicsViewAdapter::PEN_THIN_TRANSPARENT, allowdAreaFill);
235 item->setItemRole(ROLE_GRID);
236 item->setZOrder(-1);
237 }
238}

249
250void TuningCurveGraphDrawer::drawMarkers(int8_t keynumber)
251{
252 if (!mPiano) return; // no keys set yet (see setKeys)
253
254 // Draw inharmonicity levels
255 updateMarkerPosition (keynumber, ROLE_INHARMONICITY);
256
257 // Draw tuning curve
258 updateMarkerPosition (keynumber, ROLE_COMPUTED_FREQUENCY);
259
260 // Draw recorded frequency levels
261 updateMarkerPosition (keynumber, ROLE_RECORDED_FREQUENCY);
262
263 // Draw tuned frequency levels
264 updateMarkerPosition (keynumber, ROLE_TUNED_FREQUENCY);
265
266 // Draw tuned frequency levels
267 updateMarkerPosition (keynumber, ROLE_OVERPULL);
268}

290// Compute the position of a marker
291//-----------------------------------------------------------------------------
292
305
306double TuningCurveGraphDrawer::getMarkerPosition(int keyindex, RoleType role)
307{
308 // Get the key data of the piano
309 if (not mPiano) return -1;
310 const Key &key = mPiano->getKey(keyindex);
311
312 // lambda function converting a frequency ratio to cents
313 auto ratioToCents = [](double ratio)
314 { return 1200*log(ratio)/MathTools::LOG2; };
315
316 if (role & ROLE_INHARMONICITY)
317 {
318 if (mOperationMode == MODE_RECORDING)
319 {
320 double B = key.getMeasuredInharmonicity();
321 if (B > 0)return (log(B)+10.5)/8;
322 }
323 }
324 else if (role & ROLE_COMPUTED_FREQUENCY)
325 {
326 if (mOperationMode == MODE_TUNING) return 0.5; // horizontal line
327 else if (mOperationMode == MODE_CALCULATION)
328 {
329 double f = mPiano->getDefiningTempFrequency(keyindex,0,440);
330 return convertCentsToY(ratioToCents(key.getComputedFrequency() / f));
331 }
332 }
333 else if (role & ROLE_RECORDED_FREQUENCY)
334 {
335 if (mOperationMode == MODE_RECORDING or mOperationMode == MODE_IDLE)
336 {
337 double ratio = key.getRecordedFrequency() /
338 mPiano->getEqualTempFrequency(keyindex);
339 if (ratio > 0) return convertCentsToY(ratioToCents(ratio));
340 }
341 else if (mOperationMode == MODE_CALCULATION)
342 {
343 int keyA4 = mPiano->getKeyboard().getKeyNumberOfA4();
344 double fA4 = mPiano->getKey(keyA4).getRecordedFrequency();
345 double ratio = key.getRecordedFrequency() /
346 mPiano->getEqualTempFrequency(keyindex,0,fA4);
347 if (ratio > 0) return convertCentsToY(ratioToCents(ratio));
348 }
349 }
350 else if (role & ROLE_TUNED_FREQUENCY)
351 {
352 if (mOperationMode == MODE_TUNING)
353 {
354 double ratio = key.getTunedFrequency() /
355 key.getComputedFrequency() * 440.0 / mConcertPitch;
356 if (ratio > 0) return convertCentsToY(ratioToCents(ratio));
357 }
358 else if (mOperationMode == MODE_IDLE)
359 {
360 double ratio = key.getTunedFrequency() /
361 mPiano->getEqualTempFrequency(keyindex);
362 if (ratio > 0) return convertCentsToY(ratioToCents(ratio));
363 }
364 }
365 else if (role & ROLE_OVERPULL)
366 {
367 if (mOperationMode == MODE_TUNING)
368 {
369 double overpull = key.getOverpull();
370 if (fabs(overpull)>0.1) return convertCentsToY(overpull);
371 }
372 }
373 return -1;
374}

377//-----------------------------------------------------------------------------
378// Get the pen type of a marker with a given role
379//-----------------------------------------------------------------------------
380
390
391PenType TuningCurveGraphDrawer::getMarkerPenType(RoleType role)
392{
393 if (role & ROLE_INHARMONICITY) return bmarkers;
394 else if (role & ROLE_COMPUTED_FREQUENCY) return fcmarkers;
395 else if (role & ROLE_TUNED_FREQUENCY) return ftmarkers;
396 else if (role & ROLE_OVERPULL) return opmarkers;
397 else if (role & ROLE_RECORDED_FREQUENCY)
398 return (mOperationMode == MODE_CALCULATION ? fgmarkers : frmarkers);
399 else EPT_EXCEPT(EptException::ERR_NOT_IMPLEMENTED, "Color type not implemented");
400}

403//-----------------------------------------------------------------------------
404// Update the position of a marker
405//-----------------------------------------------------------------------------
406
407void TuningCurveGraphDrawer::updateMarkerPosition(int keyindex, RoleType role)
408{
409 // Compute the y coordinate where the marker should be
410 double y = getMarkerPosition(keyindex, role);
411
412 // See whether the marker already exists
413 GraphicsItem *item = mGraphics->getGraphicItem(keyindex, role);
414 if (item)
415 {
416 // if y-range valid upate position, otherwise delete it
417 if (y > 0 and y < 1)
418 item->setPosition(static_cast<double>(keyindex) / mNumberOfKeys, y);
419 else delete item;
420 }
421 else // if the marker does not exist
422 {
423 // compute x-coordinate (y-coordinate already exists)
424 double x = static_cast<double>(keyindex) / mNumberOfKeys;
425 double dx = 1.0 / mNumberOfKeys;
426 if (y > 0 && y < 1)
427 {
428 item = mGraphics->drawLine(x, y, x + dx, y, getMarkerPenType(role));
429 item->setKeyIndexAndItemRole(keyindex, role);
430 }
431 }
432}
