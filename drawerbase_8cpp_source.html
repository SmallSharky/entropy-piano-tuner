85bool DrawerBase::requestRedraw (bool force)
86{
87 auto elapsed = std::chrono::duration_cast<std::chrono::milliseconds>
88 (std::chrono::high_resolution_clock::now() - *mTimeLastDrawn);
89
90 if (force or elapsed.count() >= mRedrawIntervalInSecs * 1000)
91 {
92 *mTimeLastDrawn = std::chrono::high_resolution_clock::now();
93 return true;
94 }
95 return false;
96}
