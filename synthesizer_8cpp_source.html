100void Synthesizer::open (AudioInterface *audioInterface)
101{
102 PCMDevice::open(audioInterface);
103
104 mSampleRate = getSampleRate();
105 mChannels = getChannels();
106
107 if (mNumberOfKeys < 0 or mNumberOfKeys > 256)
108 { LogW("Called init with an invalid number of keys = %d",mNumberOfKeys); return; }
109
110 // Pre-calculate a sine wave for speedup
111 mSineWave.resize(SineLength);
112 for (int i=0; i<SineLength; ++i)
113 mSineWave[i]=static_cast<float>(sin(MathTools::TWO_PI * i / SineLength));
114
115 // Pre-calculate a piano-hammer-like noise (one second), could be improved
116 FFTComplexType amplitude (0.15/mSampleRate);
117 int datasize = std::min(mSampleRate/2+1,static_cast<int>(mHammerKnockFFT[0].size()));
118 FFTComplexVector fftl(mSampleRate/2+1,0),fftr(mSampleRate/2+1,0);
119 for (int i=0; i<datasize; i++)
120 {
121 fftl[i] = FFTComplexType(mHammerKnockFFT[0][i],-mHammerKnockFFT[1][i])*amplitude;
122 fftr[i] = FFTComplexType(mHammerKnockFFT[2][i],-mHammerKnockFFT[3][i])*amplitude;
123 }
124 FFT_Implementation transformer;
125 mHammerWaveLeft.resize(mSampleRate);
126 mHammerWaveRight.resize(mSampleRate);
127 mHammerWaveLeft.assign(mSampleRate,0);
128 mHammerWaveRight.assign(mSampleRate,0);
129 transformer.calculateFFT(fftl,mHammerWaveLeft);
130 transformer.calculateFFT(fftr,mHammerWaveRight);
131 // Note that the wave size may change if the sample rate is odd
132 // use data size instead
133 for (size_t i=0; i<mHammerWaveLeft.size(); i++)
134 {
135 mHammerWaveRight[i] *= exp(-pow(i*3.0/mHammerWaveLeft.size(),1.5));
136 mHammerWaveLeft[i] *= exp(-pow(i*3.0/mHammerWaveLeft.size(),1.5));
137 }
138
139 // Initialize reverb
140 mReverbSize = mSampleRate/10;
141 mReverbCounter = 0;
142 mReverbL.resize(mReverbSize);
143 mReverbR.resize(mReverbSize);
144 mReverbL.assign(mReverbSize,0);
145 mReverbR.assign(mReverbSize,0);
146 mDelay1 = static_cast<int> (0.1128*mReverbSize);
147 mDelay2 = static_cast<int> (0.3928*mReverbSize);
148 mDelay3 = static_cast<int> (0.8762*mReverbSize);
149
150 // Start the waveform generator
151 mWaveformGenerator.init(mNumberOfKeys,mSampleRate);
152 mWaveformGenerator.start();
153}

235void Synthesizer::playSound (const int keynumber,
236 const double frequency,
237 const double volume,
238 const Envelope &env,
239 const bool waitforcomputation,
240 const bool stereo)
241{
242 if (frequency <= 0 or volume <= 0 or mNumberOfKeys == 0) return;
243 Tone tone;
244 tone.keynumber = keynumber;
245 tone.frequency = frequency;
246 double position = (20+(keynumber&0xff)) * 1.0 / (mNumberOfKeys+40);
247 if (not stereo) position = 0.5;
248 tone.leftamplitude = sqrt((1-position)*volume);
249 tone.rightamplitude = sqrt(position*volume);
250 tone.phaseshift = (position-0.5)/500;
251 tone.envelope = env;
252 tone.clock=0;
253 tone.stage=1;
254 tone.amplitude=0;
255
256 int timeout = 0;
257 if (frequency>0 and frequency<10)
258 {
259 while (waitforcomputation and mWaveformGenerator.isComputing(keynumber)
260 and timeout++ < 1000) {
261 std::this_thread::sleep_for(std::chrono::milliseconds(1));
262 }
263 tone.waveform = mWaveformGenerator.getWaveForm(keynumber);
264 }
265 else
266 tone.waveform.clear();
267
268 mPlayingMutex.lock();
269 mPlayingTones.push_back(tone);
270 mPlayingMutex.unlock();
271}

289
290void Synthesizer::updateIntensity()
291{
292 mPlayingMutex.lock();
293 if (mPlayingTones.size()>0) mIntensity = 1;
294 mPlayingMutex.unlock();
295
296 if (mIntensity < 0.0000001)
297 {
298 mIntensity = 0;
299 }
300 else
301 {
302 // first remove all sounds with a volume below cutoff:
303 mPlayingMutex.lock();
304 for (auto it = mPlayingTones.begin(); it != mPlayingTones.end(); /* no inc */)
305 if (it->stage>=2 and it->amplitude<CutoffVolume)
306 it=mPlayingTones.erase(it);
307 else ++it;
308 //size_t N = mPlayingTones.size();
309 mPlayingMutex.unlock();
310 }
311}

335//-----------------------------------------------------------------------------
336// Generate the waveform
337//-----------------------------------------------------------------------------
338
347
348bool Synthesizer::generateAudioSignal (DataType *outputBuffer, const int64_t packet_size)
349{
350 // update intensity and played tones
351 updateIntensity();
352
353 if (mIntensity == 0 ) {
354 return false;
355 }
356
357 const int sampleRate = mSampleRate;
358 const int channels = mChannels;
359
360 int64_t clock_timeout = 40*sampleRate; // one minute timeout
361 int hammerwavesize = static_cast<int>(mHammerWaveLeft.size());
362
363 if (channels<=0 or channels>2) return false;
364
365
366 for (int bufferIndex = channels - 1; bufferIndex < packet_size; bufferIndex += channels) {
367 // pcm data
368 double left = 0, right = 0;
369
370 mPlayingMutex.lock();
371 for (Tone &tone : mPlayingTones)
372 {
373 //============== MANAGE THE ENVELOPE =================
374
375 double y = tone.amplitude; // get last amplitude
376 Envelope &envelope = tone.envelope; // get ADSR
377 switch (tone.stage) // Manage ADSR
378 {
379 case 1: // ATTACK
380 y += envelope.attack/sampleRate;
381 if (envelope.decay>0)
382 {
383 if (y >= 1) tone.stage++;
384 }
385 else
386 {
387 if (y >= envelope.sustain) tone.stage+=2;
388 }
389 break;
390 case 2: // DECAY
391 y *= (1-(1+y)*envelope.decay/sampleRate); // DECAY
392 if (y <= envelope.sustain) tone.stage++;
393 break;
394 case 3: // SUSTAIN
395 y += (envelope.sustain-y) * envelope.release / sampleRate;
396 if (tone.clock > clock_timeout) tone.stage=4;
397 break;
398 case 4: // RELEASE
399 y *= (1-envelope.release/sampleRate);
400 break;
401 }
402 tone.amplitude = y;
403 tone.clock ++;
404
405 //================ CREATE THE PCM WAVEFORM ==============
406
407 if (tone.frequency > 10) // if sine wave
408 {
409 double t = 1+tone.clock*1.0/sampleRate;
410 int i = static_cast<int64_t>(SineLength*tone.frequency*t);
411 int j = static_cast<int64_t>(SineLength*tone.frequency*(t+tone.phaseshift));
412 left += tone.leftamplitude * y * 0.2 * mSineWave[i%SineLength];;
413 right += tone.rightamplitude * y * 0.2 * mSineWave[j%SineLength];;
414 }
415 else // if complex sound
416 {
417 if (envelope.hammer) if (tone.clock < hammerwavesize)
418 {
419 left += tone.leftamplitude * mHammerWaveLeft[tone.clock];
420 int phaseshifted = static_cast<int>(tone.clock + tone.phaseshift * sampleRate);
421 if (phaseshifted > 0 and phaseshifted < hammerwavesize)
422 right += tone.rightamplitude* mHammerWaveRight[phaseshifted];
423 }
424
425 double t = (1+tone.clock*1.0/sampleRate)*tone.frequency;
426 left += tone.leftamplitude * y * 0.3 *
427 mWaveformGenerator.getInterpolation(tone.waveform,t);
428 right += tone.rightamplitude * y * 0.3 *
429 mWaveformGenerator.getInterpolation(tone.waveform,t+tone.phaseshift);
430 }
431 }
432
433 const double reverbamplitude = 0.2;
434 double echo1 = mReverbR[(mReverbCounter+mDelay1) % mReverbSize];
435 double echo2 = mReverbL[(mReverbCounter+mDelay2) % mReverbSize];
436 double echo3 = mReverbR[(mReverbCounter+mDelay3) % mReverbSize];
437 double echo4 = mReverbL[(mReverbCounter+1) % mReverbSize];
438 left += reverbamplitude * ( echo2 + echo3 );
439 right += reverbamplitude * ( echo1 + echo4 );
440 mReverbL[mReverbCounter] = left;
441 mReverbR[mReverbCounter] = right;
442 mReverbCounter = (mReverbCounter+1) % mReverbSize;
443 mIntensity = 0.98 * mIntensity + left*left + right*right;
444
445 mPlayingMutex.unlock();
446
447 // write data to buffer
448 if (channels==1)
449 {
450 outputBuffer[bufferIndex] = static_cast<DataType>((left+right)/2 * std::numeric_limits<DataType>::max());
451 }
452 else // if stereo
453 {
454 outputBuffer[bufferIndex - 1] = static_cast<DataType>(left * std::numeric_limits<DataType>::max());
455 outputBuffer[bufferIndex] = static_cast<DataType>(right * std::numeric_limits<DataType>::max());
456 }
457
458 }
459 return true;
460}

503void Synthesizer::releaseSound (const int id)
504{

506 mPlayingMutex.lock();
507 for (auto &ch : mPlayingTones) if ((ch.keynumber & 0xff)==id) { ch.stage=4; released=true; }
