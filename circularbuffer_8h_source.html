116//-----------------------------------------------------------------------------
117// Append a new data element
118//-----------------------------------------------------------------------------
119
125
126template <class data_type>
127void CircularBuffer<data_type>::push_back(const data_type &data)
128{
129 assert(mMaximumSize > 0);
130 assert(mCurrentWritePosition < mMaximumSize);
131 assert(mCurrentReadPosition < mMaximumSize);
132 // write at the desired position
133 mData[mCurrentWritePosition] = data;
134 // update write pointer
135 mCurrentWritePosition = (mCurrentWritePosition+1) % mMaximumSize;
136 // update current size
137 mCurrentSize = std::min(mCurrentSize + 1, mMaximumSize);
138 // update read pointer (add mMaximumSize to keep it positive)
139 mCurrentReadPosition = (mCurrentWritePosition + mMaximumSize - mCurrentSize) % mMaximumSize;
140}

143//-----------------------------------------------------------------------------
144// Copy data in an ordered form
145//-----------------------------------------------------------------------------
146
156
157template <class data_type>
158std::vector<data_type> CircularBuffer<data_type>::getOrderedData() const
159{
160 std::vector<data_type> data_out(mCurrentSize);
161 // copy end
162 std::size_t part1Size = std::min(mCurrentReadPosition + mCurrentSize, mMaximumSize) - mCurrentReadPosition;
163 std::memcpy(data_out.data(), mData.data() + mCurrentReadPosition, part1Size * sizeof(data_type));
164
165 // copy start
166 std::size_t part2Size = mCurrentSize - part1Size;
167 std::memcpy(data_out.data() + part1Size, mData.data(), part2Size * sizeof(data_type));
168
169 return data_out;
170}

174//-----------------------------------------------------------------------------
175// Retrieve time-ordeded data with max size of n and remove if from the buffer
176//-----------------------------------------------------------------------------
177
183
184template <class data_type>
185std::vector<data_type> CircularBuffer<data_type>::readData(size_t n)
186{
187 std::vector<data_type> data_out(std::move(getOrderedData()));
188 if (data_out.size() > n) data_out.resize(n);
189
190 EptAssert(mCurrentSize >= data_out.size(), "Do not read more data than existent.");
191
192 // reset read position
193 mCurrentReadPosition = (mCurrentReadPosition + data_out.size()) % mMaximumSize;
194 mCurrentSize = mCurrentSize - data_out.size();
195
196 return data_out;
197}

200//-----------------------------------------------------------------------------
201// Resize the buffer
202//-----------------------------------------------------------------------------
203
210
211template <class data_type>
212void CircularBuffer<data_type>::resize(std::size_t maximum_size)
213{
214 // we have to temporarily store the old data and rewrite at at position 0
215 std::vector<data_type> old_data(std::move(getOrderedData()));
216 mMaximumSize = maximum_size;
217 // new current size is size of old data, or smaller)
218 mCurrentSize = std::min(old_data.size(), maximum_size);
219 // resize or internal data structure
220 mData.resize(mMaximumSize);
221
222 // dont copy data if new buffer is empty
223 if (mMaximumSize == 0) return;
224
225 // reset read and write pointers
226 mCurrentReadPosition = 0;
227 mCurrentWritePosition = mCurrentSize % mMaximumSize;
228 // copy the old data
229 std::memcpy(mData.data(),
230 old_data.data() + (old_data.size() - mCurrentSize),
231 mCurrentSize * sizeof(data_type));
232}
