103void PitchRaise::algorithmWorkerFunction()
104{
105 LogI("Pitch Raise algorithm started");
106 // linear regression data
107 std::vector<double> X(2, 0), Y(2, 0), XX(2, 0), XY(2, 0), A(2, 0), B(2, 0);
108 std::vector<int> N(2, 0);
109
110 // make tuning curve initially flat
111 updateTuningcurve();
112
113 for (int i = 0; i < mNumberOfKeys; ++i)
114 {
115 CHECK_CANCEL_THREAD;
116 Key &key = mPiano.getKey(i);
117 showCalculationProgress(i * 0.25 / mNumberOfKeys);
118 msleep(5);
119 if (key.getMeasuredInharmonicity() > 1E-10)
120 {
121 double B = key.getMeasuredInharmonicity();
122 int s = (i < mPiano.getKeyboard().getNumberOfBassKeys() ? 0 : 1);
123 if (s == 0)
124 {
125 LogI("Found recorded key number %d in the left section with B=%lf.", i, B);
126 }
127 else
128 {
129 LogI("Found recorded key number %d in the right section with B=%lf.", i, B);
130 }
131 double x = i, y = -log(B);
132 N[s]++;
133 X[s] += x;
134 Y[s] += y;
135 XX[s] += x * x;
136 XY[s] += x * y;
137 }
138 }
139 if (N[0] < 2)
140 {
141 // Error: Not enough recorded keys in the left section
142 MessageHandler::send<MessageCaluclationProgress>(MessageCaluclationProgress::CALCULATION_FAILED,
143 MessageCaluclationProgress::CALCULATION_ERROR_NO_DATA_LEFTSECTION); // Error left
144 return;
145 }
146 if (N[1] < 2)
147 {
148 // Error: Not enough recorded keys in the right section
149 MessageHandler::send<MessageCaluclationProgress>(MessageCaluclationProgress::CALCULATION_FAILED,
150 MessageCaluclationProgress::CALCULATION_ERROR_NO_DATA_RIGHTSECTION); // Error left
151 return;
152 }
153
154 // Perform linear regression (see https://de.wikipedia.org/wiki/Lineare_Regression)
155 for (int s = 0; s < 2; ++s)
156 {
157 A[s] = (XX[s] * Y[s] - X[s] * XY[s]) / (N[s] * XX[s] - X[s] * X[s]);
158 B[s] = (N[s] * XY[s] - X[s] * Y[s]) / (N[s] * XX[s] - X[s] * X[s]);
159 }
160
161 // Define lambda regression function
162 auto estimatedInharmonicity = [this, A, B](int k)
163 {
164 int s = (k < mPiano.getKeyboard().getNumberOfBassKeys() ? 0 : 1);
165 double minusLogB = A[s] + k * B[s];
166 return exp(-minusLogB);
167 };
168
169 // For the computation we need at least two octaves to both sides of A4:
170 if (mKeyNumberOfA4 <= 13 or mNumberOfKeys - mKeyNumberOfA4 <= 13)
171 return;
172
173 // Compute the expected stretch deviation of the n_th partial of a given key:
174 auto cents = [estimatedInharmonicity](int keynumber, int n)
175 { return 600.0 / MathTools::LOG2 *
176 log((1 + n * n * estimatedInharmonicity(keynumber)) /
177 (1 + estimatedInharmonicity(keynumber))); };
178
179 // Define a linear section of the curve in the middle
180 int numberA3 = mKeyNumberOfA4 - 12;
181 int numberA4 = mKeyNumberOfA4;
182 int numberA5 = mKeyNumberOfA4 + 12;
183 double pitchA5 = 0.5 * cents(numberA4, 3) + 0.5 * cents(numberA4, 2);
184 double pitchA3 = cents(numberA4, 2) - cents(numberA3, 4);
185 for (int k = numberA3; k < mKeyNumberOfA4; ++k)
186 {
187 CHECK_CANCEL_THREAD;
188 mPitch[k] = pitchA3 * (mKeyNumberOfA4 - k) / 12.0;
189 showCalculationProgress(0.25 + (k - numberA3) * 0.125 / 12);
190 updateTuningcurve(k);
191 msleep(30);
192 }
193 for (int k = mKeyNumberOfA4 + 1; k <= numberA5; ++k)
194 {
195 CHECK_CANCEL_THREAD;
196 mPitch[k] = pitchA5 * (k - mKeyNumberOfA4) / 12.0;
197 showCalculationProgress(0.375 + (k - mKeyNumberOfA4) * 0.125 / 12);
198 updateTuningcurve(k);
199 msleep(30);
200 }
201
202 // Extend curve to the right by iteration:
203 for (int k = numberA5 + 1; k < mNumberOfKeys; k++)
204 {
205 CHECK_CANCEL_THREAD;
206 double pitch42 = mPitch[k - 12] + cents(k - 12, 4) - cents(k, 2);
207 double pitch21 = mPitch[k - 12] + cents(k - 12, 2);
208 mPitch[k] = 0.3 * pitch42 + 0.7 * pitch21;
209 updateTuningcurve(k);
210 showCalculationProgress(0.5 + (k - numberA5) * 0.25 / (mNumberOfKeys - numberA5));
211 msleep(30);
212 }
213
214 // Extend the curve to the left by iteration:
215 for (int k = numberA3 - 1; k >= 0; k--)
216 {
217 CHECK_CANCEL_THREAD;
218 double pitch42 = mPitch[k + 12] + cents(k + 12, 2) - cents(k, 4);
219 double pitch105 = mPitch[k + 12] + cents(k + 12, 5) - cents(k, 10);
220 double fraction = 1.0 * k / numberA3;
221 mPitch[k] = pitch42 * fraction + pitch105 * (1 - fraction);
222 updateTuningcurve(k);
223 msleep(30);
224 showCalculationProgress(0.75 + (numberA3 - k) * 0.25 / (numberA3));
225 }
226}
