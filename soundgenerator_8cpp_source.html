93
94void SoundGenerator::handleMessage(MessagePtr m)
95{
96 EptAssert(m, "Message has to exist!");
97
98 switch (m->getType())
99 {
100 // REFERENCE SOUND IN TUNING MODE
101 // When the key selection changes, the reference sound is stopped.
102 // In the tuning mode a new reference sound is started.
103 // This is important when the selection changes during tuning.
104 case Message::MSG_KEY_SELECTION_CHANGED:
105 {
106 // keep track of selected key
107 auto message(std::static_pointer_cast<MessageKeySelectionChanged>(m));
108 mSelectedKey = message->getKeyNumber();
109 stopResonatingReferenceSound();
110 if (mOperationMode == MODE_TUNING)
111 playResonatingReferenceSound(mSelectedKey);
112 }
113 break;
114 // START REFERENCE SOUND IN TUNING MODE AT THE BEGINNING OF RECORDING
115 case Message::MSG_RECORDING_STARTED:
116 {
117 if (mOperationMode==MODE_TUNING)
118 {
119 if (mSelectedKey>=0) playResonatingReferenceSound(mSelectedKey);
120 else stopResonatingReferenceSound();
121 }
122 }
123 break;
124 // CHANGE REFERENCE SOUND IN TUNING MODE WHEN KEY SELECTION CHANGES
125 case Message::MSG_PRELIMINARY_KEY:
126 {
127 if (mOperationMode==MODE_TUNING and mSelectedKey<0)
128 {
129 auto message(std::static_pointer_cast<MessagePreliminaryKey>(m));
130 int recognizedkey = message->getKeyNumber();
131 if (mResonatingKey != recognizedkey and recognizedkey >= 0)
132 playResonatingReferenceSound(recognizedkey);
133 }
134 }
135 break;
136 // STOP REFERENCE SOUND AT THE END OF RECORDING
137 case Message::MSG_RECORDING_ENDED:
138 {
139 // if reference tone with constant volume stop it here.
140 if (Settings::getSingleton().getSoundGeneratorMode() == SGM_REFERENCE_TONE
141 or mOperationMode!=MODE_TUNING)
142 stopResonatingReferenceSound();
143 }
144 break;
145 // REFERENCE SOUND VOLUME ADJUSTMENT IN TUNING MODE
146 case Message::MSG_RECORDER_ENERGY_CHANGED:
147 {
148 if (mOperationMode==MODE_TUNING and mResonatingKey>=0)
149 {
150 auto message(std::static_pointer_cast<MessageRecorderEnergyChanged>(m));
151 if (message->getLevelType()==MessageRecorderEnergyChanged::LevelType::LEVEL_INPUT)
152 if (Settings::getSingleton().isSoundGeneratorVolumeDynamic())
153 changeVolumeOfResonatingReferenceSound(message->getLevel());
154 }
155 }
156 break;
157 // MODE CHANGES STOP THE REFERENCE SOUND AND TRIGGER RECALCULATION
158 case Message::MSG_MODE_CHANGED:
159 {
160 // Keep track of mode changes
161 auto message(std::static_pointer_cast<MessageModeChanged>(m));
162 mOperationMode = message->getMode();
163 stopResonatingReferenceSound();
164 }
165 break;
166 // IF A NEW FILE IS OPENED OR IF PIANO SETTINGS ARE CHANGED
167 // ALL SOUNDS HAVE TO BE CALCULATED ONCE AGAIN IN THE CORRESPONDING MODE
168 case Message::MSG_PROJECT_FILE:
169 {
170 // Get a pointer to the piano and various of its properties.
171 auto mpf(std::static_pointer_cast<MessageProjectFile>(m));
172 mPiano = &(mpf->getPiano());
173 mNumberOfKeys = mPiano->getKeyboard().getNumberOfKeys();
174 mSynthesizer.setNumberOfKeys(mNumberOfKeys);
175 mKeyNumberOfA4 = mPiano->getKeyboard().getKeyNumberOfA4();
176 stopResonatingReferenceSound();
177 preCalculateSoundOfAllKeys();
178 }
179 break;
180 // HANDLE MIDI KEYPRESSES AND RELATED EVENTS
181 case Message::MSG_MIDI_EVENT:
182 {
183 auto message(std::static_pointer_cast<MessageMidiEvent>(m));
184 MidiAdapter::Data data = message->getData();
185 switch (data.event)
186 {
187 case MidiAdapter::MIDI_KEY_PRESS:
188 {
189 if (data.byte2 == 0) {
190 // 0 volume will be handled as release
191 int key = data.byte1-69+mKeyNumberOfA4;
192 mSynthesizer.releaseSound(key);
193 } else {
194 handleMidiKeypress(data); // see following function
195 }
196 break;
197 }
198 case MidiAdapter::MIDI_KEY_RELEASE:
199 {
200 int key = data.byte1-69+mKeyNumberOfA4;
201 mSynthesizer.releaseSound(key);
202 break;
203 }
204 case MidiAdapter::MIDI_CONTROL_CHANGE:
205 {
206// // Funny feature that allows you to switch
207// // between the operating modes by MIDI pedal
208// // If it is not a pedal break (damper=67):
209// if (data.byte1 < 64 or data.byte1 > 67) break;
210// // If pedal pressed go to recording mode.
211// if (data.byte2 > 0)
212// MessageHandler::send<MessageModeChanged>(MODE_RECORDING);
213// else // else go back to calculation mode
214// MessageHandler::send<MessageModeChanged>(MODE_CALCULATION);
215 }
216 default:
217 break;
218 }
219 }
220 break;
221 // RECALCULATE AND PLAY ECHO SOUND IN THE RECORDING MODE
222 case Message::MSG_FINAL_KEY:
223 {
224 // echo synthesized sound of the recorded key in recording mode.
225 if (mOperationMode==MODE_RECORDING)
226 {
227 auto message(std::static_pointer_cast<MessageFinalKey>(m));
228 int keynumber = message->getKeyNumber();
229 auto spectrum = message->getFinalKey()->getPeaks();
230 // replay only if the selected key was recognized:
231 if (keynumber == mSelectedKey and spectrum.size() > 0)
232 {
233 preCalculateSoundOfKey (keynumber,spectrum);
234 mSynthesizer.playSound(keynumber,1,0.2,Envelope(5,5,0,30,false),true,false);
235 }
236 }
237 }
238 break;
239 // RECALCULTE WAVEFORM DURING CALCULATION MODE WHEN FREUQUENCY CHANGES
240 case Message::MSG_CHANGE_TUNING_CURVE:
241 {
242 if (mOperationMode==MODE_CALCULATION)
243 {
244 auto message(std::static_pointer_cast<MessageChangeTuningCurve>(m));
245 preCalculateSoundOfKey(message->getKeyNumber());
246 }
247 }
248 break;
249 default:
250 break;
251 }
252}

269
270void SoundGenerator::handleMidiKeypress (MidiAdapter::Data &data)
271{
272 int key = data.byte1-69+mKeyNumberOfA4; // extract key number starting with 0
273 if (key<0 or key>=mNumberOfKeys) return;
274 double volume = pow(static_cast<double>(data.byte2) / 128, 2); // keystroke volume
275 // The following formula mimics the decay time of a piano string
276 const double decay = (key <= 12 ? 1.0/6 : 1.0/210*pow(key,1.43));
277 const double release = (key - mKeyNumberOfA4 >= 22 ? decay : 30);
278 Envelope envelope (40,decay,0,release,true);
279
280 switch(mOperationMode)
281 {
282 case MODE_IDLE:
283 {
284 // In this mode play sine waves with respect to the selected concert pitch
285 double frequency = mPiano->getEqualTempFrequency(key,0,mPiano->getConcertPitch());
286 mSynthesizer.playSound(key,frequency,volume,Envelope(40,5,0.6,10));
287 }
288 break;
289 case MODE_RECORDING:
290 {
291 // In this mode select key and play the original sound in the original pitch
292 MessageHandler::send<MessageKeySelectionChanged>(key, &(mPiano->getKey(key)));
293 mSynthesizer.playSound(key,1,0.1*volume,envelope);
294 }
295 break;
296 case MODE_CALCULATION:
297 {
298 // In this mode select key and play the original sound in the original pitch
299 MessageHandler::send<MessageKeySelectionChanged>(key, &(mPiano->getKey(key)));
300 double recorded = mPiano->getKey(key).getRecordedFrequency();
301 if (recorded > 0)
302 {
303 double frequencyshift = mPiano->getKey(key).getComputedFrequency() *
304 mPiano->getConcertPitch() / 440.0 / recorded;
305 mSynthesizer.playSound(key,frequencyshift,0.1*volume,envelope,true);
306 }
307 }
308 break;
309 case MODE_TUNING:
310 {
311 // In this mode play the computed sound in selected concert pitch
312 double recorded = mPiano->getKey(key).getRecordedFrequency();
313 if (recorded > 0)
314 {
315 double frequencyshift = mPiano->getKey(key).getComputedFrequency() *
316 mPiano->getConcertPitch() / 440.0 / recorded;
317 mSynthesizer.playSound(key,frequencyshift,0.1*volume,envelope);
318 }
319 }
320 break;
321 default:
322 break;
323 }
324}

345
346void SoundGenerator::playResonatingReferenceSound (int keynumber)
347{
348 if (mResonatingKey >= 0 and keynumber != mResonatingKey)
349 stopResonatingReferenceSound();
350 if (keynumber < 0 or keynumber >= mNumberOfKeys) return;
351 if (mSynthesizer.isPlaying(keynumber)) return;
352 auto &key = mPiano->getKey(keynumber);
353 double frequency = key.getComputedFrequency()*mPiano->getConcertPitch()/440.0;
354 if (frequency>0)
355 {
356 mResonatingKey=keynumber;
357 mResonatingVolume = 1;
358 switch (Settings::getSingleton().getSoundGeneratorMode())
359 {
360 case SGM_REFERENCE_TONE:
361 {
362 playResonatingSineWave(keynumber,frequency, 0.5);
363 }
364 break;
365 case SGM_SYNTHESIZE_KEY:
366 {
367 const double volume = 0.2;
368 Envelope env(30,50,mResonatingVolume,10,false);
369 double frequencyshift = frequency /
370 mPiano->getKey(keynumber).getRecordedFrequency();
371 mSynthesizer.playSound(keynumber,frequencyshift,volume,env,false,false);
372 }
373 break;
374 default:
375 break;
376 }
377 }

401//-----------------------------------------------------------------------------
402// Change the volume of the resonating reference sound
403//-----------------------------------------------------------------------------
404
415
416void SoundGenerator::changeVolumeOfResonatingReferenceSound (double level)
417{
418 if (mResonatingKey < 0 or mResonatingKey >= mNumberOfKeys) return;
419 if (not mSynthesizer.isPlaying(mResonatingKey)) { mResonatingKey=-1; return; }
420 double truncatedlevel = std::min(0.8,level);
421 double volume = pow(truncatedlevel,2.0);
422 if (volume > mResonatingVolume) mResonatingVolume = volume;
423 else mResonatingVolume *= 0.87;
424 mSynthesizer.ModifySustainLevel(mResonatingKey,mResonatingVolume);
425}

428//-----------------------------------------------------------------------------
429// Play a permanent sine wave as a resonating background sound for tuning
430//-----------------------------------------------------------------------------
431
438
439void SoundGenerator::playResonatingSineWave (int keynumber, double frequency, double volume)
440{
441
442 auto &key = mPiano->getKey(keynumber);
443 auto &peaks = key.getPeaks();
444 if (peaks.size()==0 or frequency==0 or volume==0) return;
445 if (keynumber < mKeyNumberOfA4 and peaks.size()<4) return;
446 auto it = peaks.begin();
447 double factor = frequency / it->first;
448 if (keynumber < mKeyNumberOfA4-24) it++;
449 if (keynumber < mKeyNumberOfA4-30) it++;
450 if (keynumber < mKeyNumberOfA4-36) it++;
451 double frequ = it->first*factor;
452 mSynthesizer.playSound(keynumber,frequ,0.5,Envelope(30,5,1,30),false,false);
453}
