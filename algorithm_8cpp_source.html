56
57void Algorithm::workerFunction_impl()
58{
59 setThreadName("Algorithm");
60 LogI("Start calculation.");
61 MessageHandler::send<MessageCaluclationProgress>
62 (MessageCaluclationProgress::CALCULATION_STARTED);
63
64 // Call the worker function of the algorithm from here
65 algorithmWorkerFunction();
66
67 // After completion of the algorithm, deselect all keys.
68 MessageHandler::send<MessageKeySelectionChanged>(-1,nullptr);
69
70 MessageHandler::send<MessageCaluclationProgress>
71 (MessageCaluclationProgress::CALCULATION_ENDED);
72 LogI("End of calculation");
73}
