30std::shared_ptr<const AlgorithmInformation> AlgorithmInformationParser::parse(const AlgorithmFactoryDescription &algorithmFactory) const {
31 const std::string &algorithmId(algorithmFactory.getAlgorithmName());
32 const std::string &algorithmVersion(algorithmFactory.getVersion());
33
34 XmlReaderInterfacePtr readerPtr = XmlFactory::getDefaultReader();
35 XmlReaderInterface &reader = *readerPtr;
36
37 std::wstring fileContent(FileManager::getSingleton().getAlgorithmInformationFileContent(algorithmId));
38
39 reader.openString(fileContent);
40
41 std::wstring author, name, year, description;
42 AlgorithmInformation::ParameterListType params;
43
44 while (!reader.atEnd()) {
45 reader.readNext();
46
47 if (!reader.isStartElement()) {continue;}
48
49 if (reader.name() == "algorithm") {
50 // root node content
51 year = reader.queryWStringAttribute("year");
52 author = reader.queryWStringAttribute("author");
53 } else if (reader.name() == "name") {
54 name = parseLanguageString(reader);
55 } else if (reader.name() == "description") {
56 description = parseLanguageString(reader);
57 } else if (reader.name() == "param") {
58 params.push_back(parseAlgorithmParameter(reader));
59 }
60 }
61
62 return std::shared_ptr<const AlgorithmInformation>(
63 new AlgorithmInformation(algorithmId, algorithmVersion, name, description, year, author, params));
64}

66std::wstring AlgorithmInformationParser::parseLanguageString(XmlReaderInterface &reader) const {
67 std::string languageId = Settings::getSingleton().getUserLanguageId();
68
69 std::wstring defaultLanguageString;
70
71 const std::string currentNodeName = reader.name();
72 while (!reader.atEnd()) {
73 reader.readNext();
74 if (reader.isEndElement() && reader.name() == currentNodeName) {break;}
75 if (reader.isStartElement() == false) {continue;}
76
77 if (reader.name() != "string") {
78 EPT_EXCEPT(EptException::ERR_INVALIDPARAMS, "Only string names are expected when parsing elements of type '"
79 + currentNodeName + "', but element with name '"
80 + reader.name() + "' was given.");
81 }
82
83 if (reader.hasAttribute("lang")) {
84 if (languageId == reader.queryStringAttribute("lang")) {
85 return reader.queryWStringText();
86 }
87 } else {
88 // default
89 defaultLanguageString = reader.queryWStringText();
90
91 if (languageId == "en") {
92 // default language is en
93 return defaultLanguageString;
94 }
95 }
96
97 // not the desired string
98 }
99
100 // language string not found. Check if default string exist, then return this one
101 if (defaultLanguageString.size() > 0) {
102 LogD("Element '%s' has missing translation for '%s'", currentNodeName.c_str(), languageId.c_str());
103 return defaultLanguageString;
104 }
105
106 EPT_EXCEPT(EptException::ERR_INVALIDPARAMS, "Element '" + currentNodeName + "' has no child elements with name 'string'");
107}

109AlgorithmParameterDescription AlgorithmInformationParser::parseAlgorithmParameter(XmlReaderInterface &reader) const {
110
111 const std::string id = reader.queryStringAttribute("id");
112 const std::string type = reader.queryStringAttribute("type");
113
114 std::wstring label, description;
115
116 AlgorithmParameterDescription parameter;
117 AlgorithmParameterDescription::StringParameterList &list = parameter.getStringList();
118
119 parameter.getID() = id;
120
121 // nonspecific parameters
122 reader.queryBoolAttributeRef("lineEdit", parameter.displayLineEdit());
123 reader.queryBoolAttributeRef("spinBox", parameter.displaySpinBox());
124 reader.queryBoolAttributeRef("slider", parameter.displaySlider());
125 reader.queryBoolAttributeRef("defaultButton", parameter.displaySetDefaultButton());
126 reader.queryBoolAttributeRef("readOnly", parameter.readOnly());
127 reader.queryIntAttributeRef("updateInterval", parameter.updateIntervalInMS());
128
129 // specific parameters
130 if (type == "double") {
131 parameter.setType(AlgorithmParameterDescription::TYPE_DOUBLE);
132
133 reader.queryRealAttributeRef("default", parameter.getDoubleDefaultValue());
134 reader.queryRealAttributeRef("min", parameter.getDoubleMinValue());
135 reader.queryRealAttributeRef("max", parameter.getDoubleMaxValue());
136 reader.queryIntAttributeRef("precision", parameter.getDoublePrecision());
137 } else if (type == "int") {
138 parameter.setType(AlgorithmParameterDescription::TYPE_INT);
139
140 reader.queryIntAttributeRef("default", parameter.getIntDefaultValue());
141 reader.queryIntAttributeRef("min", parameter.getIntMinValue());
142 reader.queryIntAttributeRef("max", parameter.getIntMaxValue());
143 } else if (type == "list") {
144 parameter.setType(AlgorithmParameterDescription::TYPE_LIST);
145
146 parameter.getStringDefaultValue() = reader.queryStringAttribute("default");
147 } else if (type == "bool" || type == "boolean") {
148 parameter.setType(AlgorithmParameterDescription::TYPE_BOOL);
149
150 reader.queryBoolAttributeRef("default", parameter.getBoolDefaultValue());
151 } else {
152 EPT_EXCEPT(EptException::ERR_INVALIDPARAMS, "Parameter type '" + type + "' is not supported.");
153 }
154
155
156 const std::string currentName = reader.name();
157
158 while (reader.atEnd() == false) {
159 reader.readNext();
160
161 if (reader.isEndElement() && reader.name() == currentName) {break;}
162 if (!reader.isStartElement()) {continue;}
163
164 if (reader.name() == "label") {
165 parameter.getLabel() = parseLanguageString(reader);
166 } else if (reader.name() == "description") {
167 parameter.getDescription() = parseLanguageString(reader);
168 } else if (reader.name() == "entry") {
169 std::string value = reader.queryStringAttribute("value");
170 std::wstring label = parseLanguageString(reader);
171 list.push_back(std::make_pair(value, label));
172 }
173 }
174
175
176
177 EptAssert(parameter.getType() != AlgorithmParameterDescription::TYPE_UNSET, "The parameter type was not set");
178 EptAssert(parameter.getID().empty() == false, "The parameter id is unset");
179
180 return parameter;
181}
