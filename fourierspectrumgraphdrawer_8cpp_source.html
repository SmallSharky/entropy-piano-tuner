93void FourierSpectrumGraphDrawer::updateSpectrum()
94{
95 // delete old peaks and chart
96 // ==========================================================================
97
98 GraphicsItem *item = mGraphics->getGraphicItemByRole(ROLE_CHART);
99 if (item)
100 {
101 delete item;
102 item = nullptr;
103 }
104
105 GraphicItemsList peakItems = mGraphics->getGraphicItemsByRole(ROLE_PEAK);
106 for (GraphicsItem *peakItem : peakItems)
107 {
108 delete peakItem;
109 }
110
111 // draw new peaks and chart
112 // ==========================================================================
113
114
115 // check if data is available, otherwise quit
116 if (not mPolygon) return;
117
118 // lambda function mapping the frequency to an x-coodinate in [0,1]:
119 const double a = (mKeyNumberOfA4+0.5)/mNumberOfKeys;
120 const double b = 12.0 / mNumberOfKeys / MathTools::LOG2;
121 auto xposition = [a,b,this] (double f) { return a + b*log(f/mConcertPitch); };
122
123
124 // MARK PEAKS AS THIN BLUE DOTS IN THE BACKGROUND (mKey needed)
125
126 const double exponent = 0.3; // magnify low power spectral lines nonlinearly
127
128 if (mCurrentOperationMode == MODE_RECORDING) if (mKey)
129 {
130 Key::PeakListType peaks = mKey->getPeaks();
131 for (auto p : peaks)
132 {
133 double x = xposition(p.first);
134
135 // Search for the corresponding peaks in the polygon:
136 auto pos1 = mPolygon->begin();
137 while (pos1!=mPolygon->end() and pos1->first < p.first*0.995) pos1++;
138 auto pos2 = pos1;
139 while (pos2!=mPolygon->end() and pos2->first < p.first*1.005) pos2++;
140 auto comp = [] (const std::pair<double,double> &a, const std::pair<double,double> &b)
141 { return a.second < b.second; };
142 auto maxelem = std::max_element(pos1,pos2, comp);
143 if (maxelem != mPolygon->end())
144 {
145 double y = 1-0.95*pow(maxelem->second, exponent);
146 const double dx=0.003;
147 const double dy=0.02;
148 item = mGraphics->drawFilledRect(x-dx/2,y-dy/2,dx,dy,
149 GraphicsViewAdapter::PEN_THIN_TRANSPARENT,
150 GraphicsViewAdapter::FILL_BLUE);
151 if (item) item->setItemRole(ROLE_PEAK);
152 }
153 }
154 }
155
156 // DRAW SPECTRUM
157
158 std::vector<GraphicsViewAdapter::Point> points;
159 EptAssert(mConcertPitch > 0,"concert pitch should be positive");
160 EptAssert(mNumberOfKeys > 0,"invalid number of keys");
161
162 for (auto &p : *mPolygon)
163 {
164 double x=xposition(p.first);
165 if (x>=0 and x<=1) points.push_back({x, 1-0.95*pow(p.second,exponent)});
166 }
167 item = mGraphics->drawChart(points, GraphicsViewAdapter::PEN_THIN_RED);
168 if (item) item->setItemRole(ROLE_CHART);
169}

191void FourierSpectrumGraphDrawer::handleMessage(MessagePtr m)
192{
193 switch (m->getType())
194 {
195 case Message::MSG_PROJECT_FILE:
196 {
197 // Update local parameter
198 auto mpf(std::static_pointer_cast<MessageProjectFile>(m));
199 mConcertPitch = mpf->getPiano().getConcertPitch();
200 mNumberOfKeys = mpf->getPiano().getKeyboard().getNumberOfKeys();
201 mKeyNumberOfA4 = mpf->getPiano().getKeyboard().getKeyNumberOfA4();
202 mPolygon.reset();
203 redraw(true);
204 break;
205 }
206 case Message::MSG_MODE_CHANGED:
207 {
208 // Update operation mode
209 auto mmc(std::static_pointer_cast<MessageModeChanged>(m));
210 mCurrentOperationMode = mmc->getMode();
211 break;
212 }
213 case Message::MSG_NEW_FFT_CALCULATED:
214 {
215 // Update mPolygon
216 auto mnfc(std::static_pointer_cast<MessageNewFFTCalculated>(m));
217 if (mnfc->hasError())
218 {
219 mPolygon.reset();
220 }
221 else
222 {
223 mPolygon = mnfc->getPolygon();
224 mSamplingRate = mnfc->getData()->samplingRate;
225 }
226 // Reset mKey
227 mKey.reset();
228 // Plot curve
229 if (requestRedraw(mnfc->isFinal())) updateSpectrum();
230 break;
231 }
232 case Message::MSG_CLEAR_RECORDING:
233 // Clear everything, cancel spectrum
234 mPolygon.reset();
235 mKey.reset();
236 redraw(true);
237 break;
238 case Message::MSG_CALCULATION_PROGRESS:
239 redraw();
240 break;
241 case Message::MSG_FINAL_KEY:
242 {
243 // Update mKey
244 auto mnfc(std::static_pointer_cast<MessageFinalKey>(m));
245 mKey = mnfc->getFinalKey();
246 if (requestRedraw(true)) updateSpectrum();
247 break;
248 }
249 default:
250 break;
251 }
252}
