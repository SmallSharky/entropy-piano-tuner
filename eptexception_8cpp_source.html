44const std::string& EptException::getFullDescription(void) const
45{
46 if (mFullDesc.empty())
47 {
48
49 std::stringstream desc;
50
51 desc << "EPT EXCEPTION(" << mNumber << ":" << mTypeName << "): "
52 << mDescription
53 << " in " << mSource;
54
55 if( mLine > 0 )
56 {
57 desc << " at " << mFile << " (mLine " << mLine << ")";
58 }
59
60 mFullDesc = desc.str();
61 }
62
63 return mFullDesc;
64}
