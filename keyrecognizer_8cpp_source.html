111void KeyRecognizer::recognizeKey (
112 bool forceRestart,
113 const Piano *piano,
114 FFTDataPointer fftPointer,
115 int selectedKey,
116 bool keyForced)
117{
118 EptAssert(piano, "The piano has to be set.");
119 EptAssert(fftPointer, "The fft data has to exist.");
120 EptAssert(fftPointer->isValid(), "Invaild fft data");
121
122 if (forceRestart) stop(); // if restart forced stop thread
123 else if (isThreadRunning()) return; // if it is running do nothing
124
125 // copy data from the piano
126 mPiano = piano;
127 mConcertPitch = piano->getConcertPitch();
128 mNumberOfKeys = piano->getKeyboard().getNumberOfKeys();
129 mKeyNumberOfA = piano->getKeyboard().getKeyNumberOfA4();
130 mFFTPtr = fftPointer; // save pointer to the Fourier transform
131 mSelectedKey = selectedKey; // copy selected key
132 mKeyForced = keyForced; // copy forcing flag
133
134 start(); // start the thread
135}

138//-----------------------------------------------------------------------------
139// Worker function running in an independent thread
140//-----------------------------------------------------------------------------
141
145
146void KeyRecognizer::workerFunction()
147{
148 setThreadName("KeyRecognizer");
149 EptAssert(mFFTPtr, "FFT Data have to non zero");
150 EptAssert(mFFTPtr->isValid(), "FFT Data have to exist");
151 EptAssert(mCallback, "Callback class has to exist");
152
153 double f=0;
154 if (mKeyForced and mSelectedKey>=0) f = detectForcedFrequency();
155 else f = detectFrequencyInTreble();
156 CHECK_CANCEL_THREAD;
157
158 if (f==0)
159 {
160 constructLogSpec(); // Compute log spectrum
161 CHECK_CANCEL_THREAD;
162
163 signalPreprocessing(); // Compute flattened loglog spectrum
164 CHECK_CANCEL_THREAD;
165
166 f = estimateFrequency(); // Estimate frequency
167 CHECK_CANCEL_THREAD;
168 }
169
170 int keynumber = findNearestKey(f); // determine keynumber
171
172 mCallback->keyRecognized(keynumber, f); // notify callback
173}

176//-----------------------------------------------------------------------------
177// Measure forced keys
178//-----------------------------------------------------------------------------
179
187
188double KeyRecognizer::detectForcedFrequency()
189{
190 if (mSelectedKey<0 or not mKeyForced) return 0;
191 std::vector<double> &fft = mFFTPtr->fft;
192 const int n = static_cast<int>(mFFTPtr->fft.size());
193 const int sr = mFFTPtr->samplingRate;
194 auto ftoq = [n,sr] (double f) { return MathTools::roundToInteger(2*n*f/sr); };
195 auto qtof = [n,sr] (double q) { return sr*q/(2*n); };
196 double f = mPiano->getEqualTempFrequency(mSelectedKey);
197 const int q1 = std::max(0,ftoq(f/1.04));
198 const int q2 = std::min(ftoq(f*1.04),n);
199 double max=0;
200 for (int q=q1; q<=q2; ++q) if (fft[q]>max)
201 {
202 max=fft[q];
203 f=qtof(q);
204 }
205 return f;
206}

209//-----------------------------------------------------------------------------
210// Detect treble keys
211//-----------------------------------------------------------------------------
212
223
224double KeyRecognizer::detectFrequencyInTreble()
225{
226 const double threshold = 0.09;
227 std::vector<double> &fft = mFFTPtr->fft;
228 const int n = static_cast<int>(mFFTPtr->fft.size());
229 const int sr = mFFTPtr->samplingRate;
230 auto ftoq = [n,sr] (double f) { return MathTools::roundToInteger(2*n*f/sr); };
231 auto qtof = [n,sr] (double q) { return sr*q/(2*n); };
232 const int q1=ftoq(20), q2=ftoq(1000), q3=ftoq(4500);
233 double m1=0, m2=0;
234 if (n<=0 or q1<0 or q3>=n) return 0;
235 for (int q=q1; q<=q3; ++q)
236 {
237 m1 += fft[q]*q;
238 m2 += fft[q]*q*q;
239 }
240 if (m1<=0) return 0;
241 double f=0;
242 if (m2/m1/n > threshold)
243 {
244 double max=0;
245 for (int q=q2; q<q3; ++q) if (fft[q]>max)
246 {
247 max=fft[q];
248 f=qtof(q);
249 }
250 }
251 return f;
252}

271int KeyRecognizer::ftom (double f) {
272 return static_cast<int>(0.5+M*(log(f)-logfmin)/(logfmax-logfmin));

307void KeyRecognizer::constructLogSpec()
308{
309 const int Q = static_cast<int>(mFFTPtr->fft.size());
310 std::function<double(double)> mtoq = [this,Q] (double m)
311 { return 2*fmin*Q/mFFTPtr->samplingRate*pow(fmax/fmin,m/M); };
312 MathTools::coarseGrainSpectrum (mFFTPtr->fft,mLogSpec,mtoq);

316//-----------------------------------------------------------------------------
317// Preprocessing of the singal
318//-----------------------------------------------------------------------------
319
334
335void KeyRecognizer::signalPreprocessing()
336{
337 Write("01-logspec.dat",mLogSpec);
338 double norm = MathTools::computeNorm(mLogSpec);
339 if (norm<=0) return;
340 auto decibel = [norm](double x) {return 10*log10(x/norm);};
341 auto dB= MathTools::transformVector<double>(mLogSpec,decibel);
342 Write("02-dB.dat",dB);
343
344
345 // Flatten noise by subtracting a gliding average
346 for (int i=0; i<M; ++i)
347 {
348 //const int w=MathTools::roundToInteger(1+5000/mtof(i)); // width of the average
349 const int w=30;
350 int a = std::max(0,i-w);
351 int b = std::min(M,i+1);
352 double sum=0;
353 for (int k=a; k<b; ++k) sum+=dB[k]*dB[k];
354 mFlatSpectrum[i]=std::max(0.0,dB[i]+sqrt(sum/(b-a))-5);
355 }
356 Write("03-dBflat.dat",mFlatSpectrum);
357
358
359 // If key is selected decrease all spectral lines below
360 // and slightly increase the main spectral line.
361 if (mSelectedKey>=0 and not mKeyForced)
362 {
363 int m1=ftom(mPiano->getEqualTempFrequency(mSelectedKey)*0.93);
364 int m2=ftom(mPiano->getEqualTempFrequency(mSelectedKey)*1.07);
365 if (m1>=0 and m1<=m2 and m2<=M)
366 {
367 for (int k=0; k<m1; ++k) mFlatSpectrum[k] *= 0.75;
368 for (int k=m1; k<m2; ++k) mFlatSpectrum[k] *= 1.2;
369 }
370 }
371 Write("04-FlatSpectrum.dat",mFlatSpectrum);
372
373}

376//-----------------------------------------------------------------------------
377// Define the kernel
378//-----------------------------------------------------------------------------
379
390
391void KeyRecognizer::defineKernel ()
392{
393
394 static const int width=M/300; // width of the peaks in bins
395 static const int partials=20; // number of partials to be detected (20)
396 static const double B=0.000; // here still with a constant inharmonicity
397
398 static std::vector<double> kernel(M); //must be static for using fftw3
399 kernel.assign(M,0);
400
401 // lambda function for setting a peak
402 auto setpeak = [] (int m, double amplitude)
403 { for (int n=m-width; n<=m+width; n++) kernel[(n+M)%M]=amplitude*(width-std::abs(n-m)); };
404
405 // lambda function for computing the frequency ratio of the nth partial
406 auto partial = [] (int n, double B)
407 { return n * sqrt((1+B*n*n)/(1+B)); };
408
409 // lambda function for computing the index of the nth partial
410 auto partialindex = [this,partial] (int n, double B, int div)
411 { return ftom(fmin*partial(n,B)/partial(div,B)); };
412
413 // lambda function for the intensity decay of the peaks
414 auto intensity = [] (int n) { return pow(static_cast<double>(n),-0.3); };
415
416 // Define the kernel function
417 for (int div=2; div<=4; div++) for (int n=1; n<=30; ++n) if (n%div>0) if (n>div-2)
418 setpeak(partialindex(n,B,div),-0.3*intensity(n));
419
420 for (int n=1; n<=partials; ++n) setpeak(partialindex(n, B, 1),intensity(n));
421
422
423 mFFT.calculateFFT(kernel,mKernelFFT); // calculate FFT
424
425 Write("05-keyrecog-kernel.dat",kernel,true);
426}

429//-----------------------------------------------------------------------------
430// Estimate frequency
431//-----------------------------------------------------------------------------
432
445
446double KeyRecognizer::estimateFrequency ()
447{
448 mFFT.calculateFFT(mFlatSpectrum,mFlatFFT);
449 for (size_t n = 0; n < mFlatFFT.size(); ++n)
450 mFlatFFT[n] *= std::conj(mKernelFFT[n]);
451 mFFT.calculateFFT(mFlatFFT,mConvolution);
452 int m = MathTools::findMaximum(mConvolution);
453 Write("06-convolution.dat",mConvolution,false);
454 return mtof(m);
455}

458//-----------------------------------------------------------------------------
459// Find the nearest key to a given frequency, assuming average stretch
460//-----------------------------------------------------------------------------
461
470
471int KeyRecognizer::findNearestKey (double f)
472{
473 if (mConcertPitch<=390 or mConcertPitch>500) return -1;
474 // Approximate distance in keys from A-440:
475 double d = 17.3123*log(f/mConcertPitch);
476 // Average stretch polynomial, giving the expected deviation in cents
477 double c = 0.000019394+0.079694594*d-0.003718646*d*d
478 + 0.000450934*d*d*d + 0.000003724*d*d*d*d;
479 int k=-1; k = static_cast<int>(mKeyNumberOfA+d-c/100+0.5);
480 return (k>=0 and k<mNumberOfKeys ? k : -1);
481}
