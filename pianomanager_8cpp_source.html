74void PianoManager::resetPitches()
75{
76 switch (mOperationMode)
77 {
78 case MODE_IDLE:
79 case MODE_RECORDING:
80 mPiano.getKeyboard().clearAllKeys();
81 break;
82 case MODE_CALCULATION:
83 mPiano.getKeyboard().clearComputedPitches();
84 mPiano.getKeyboard().clearOverpulls();
85 break;
86 case MODE_TUNING:
87 mPiano.getKeyboard().clearTunedPitches();
88 mPiano.getKeyboard().clearOverpulls();
89 break;
90 default:
91 break;
92 }
93 MessageHandler::send(Message::MSG_CLEAR_RECORDING);
94}

97//-----------------------------------------------------------------------------
98// Message listener and dispatcher
99//-----------------------------------------------------------------------------
100
105
106void PianoManager::handleMessage(MessagePtr m)
107{
108 EptAssert(m, "Message has to exist!");
109
110 switch (m->getType())
111 {
112 case Message::MSG_PROJECT_FILE:
113 {
114 auto message(std::static_pointer_cast<MessageProjectFile>(m));
115 switch (message->getFileMessageType())
116 {
117 case MessageProjectFile::FILE_CREATED:
118 case MessageProjectFile::FILE_OPENED:
119 // on open or create (new file) we select the next nex
120 MessageHandler::sendUnique<MessageKeySelectionChanged>(findNextKey(-1), nullptr);
121 break;
122 default:
123 break;
124 }
125 break;
126 }
127 case Message::MSG_MODE_CHANGED:
128 {
129 auto message(std::static_pointer_cast<MessageModeChanged>(m));
130 mOperationMode = message->getMode();
131 if (mOperationMode==MODE_RECORDING)
132 {
133 int nextkey = findNextKey(-1);
134 MessageHandler::send<MessageKeySelectionChanged>(nextkey, mPiano.getKeyPtr(nextkey));
135 }
136 }
137 break;
138 case Message::MSG_KEY_SELECTION_CHANGED:
139 {
140 auto message(std::static_pointer_cast<MessageKeySelectionChanged>(m));
141 mSelectedKey = message->getKeyNumber();
142 mForcedRecording = message->isForced();
143 }
144 break;
145 case Message::MSG_FINAL_KEY:
146 {
147 auto message(std::static_pointer_cast<MessageFinalKey>(m));
148 auto keyptr = message->getFinalKey(); // get shared pointer to the new key
149 int keynumber = message->getKeyNumber();
150 EptAssert(keynumber >= 0 && keynumber < mPiano.getKeyboard().getNumberOfKeys(),"Range of keynumber");
151 handleNewKey(keynumber,keyptr);
152 }
153 break;
154 case Message::MSG_CHANGE_TUNING_CURVE:
155 {
156 auto message(std::static_pointer_cast<MessageChangeTuningCurve>(m));
157 int keynumber = message->getKeyNumber();
158 double frequency = message->getFrequency();
159 EptAssert(keynumber >= 0 and keynumber < mPiano.getKeyboard().getNumberOfKeys(), "range of keynumber");
160 mPiano.getKey(keynumber).setComputedFrequency(frequency);
161 MessageHandler::send<MessageKeyDataChanged>(keynumber, mPiano.getKeyPtr(keynumber));
162 }
163 break;
164 default:
165 break;
166 }
167}

169//-----------------------------------------------------------------------------
170// Take action of a new key has been identified
171//-----------------------------------------------------------------------------
172
181
182void PianoManager::handleNewKey (int keynumber, std::shared_ptr<Key> keyptr)
183{
184 if (mOperationMode == MODE_RECORDING)
185 {
186 if (keynumber==mSelectedKey or mForcedRecording)
187 {
188 std::cout << "PianoManager: Sucessfully inserted new key spectrum" << std::endl;
189 mPiano.setKey(mSelectedKey,*keyptr);
190 // notify, that key data changed (e.g. tuning curve will redraw)
191 MessageHandler::send<MessageKeyDataChanged>(mSelectedKey, mPiano.getKeyPtr(mSelectedKey));
192
193#if CONFIG_ENABLE_XMGRACE
194 std::ofstream os ("4-quality.dat");
195 for (auto &p : mPiano.getKeyboard().getKeys())
196 os << p.getRecognitionQuality() << std::endl;
197 os.close();
198#endif
199 }
200 }
201 else if (mOperationMode == MODE_TUNING)
202 {
203 double frequency = keyptr->getTunedFrequency();
204 double overpull = keyptr->getOverpull();
205 double tuned = keyptr->getTunedFrequency();
206 Key* keypointer = mPiano.getKeyPtr(keynumber);
207 if (keynumber==mSelectedKey or mForcedRecording)
208 keypointer->setTunedFrequency(frequency);
209 keypointer->setOverpull(overpull);
210 keypointer->setTunedFrequency(tuned);
211 MessageHandler::send<MessageKeyDataChanged>(keynumber, keypointer);
212
213 }
214}

217//-----------------------------------------------------------------------------
218// Find next key
219//-----------------------------------------------------------------------------
220
229
230int PianoManager::findNextKey (int keynumber) // -1 if none
231{
232 const int K = mPiano.getKeyboard().getNumberOfKeys();
233 if (keynumber < K-1)
234 {
235 if (mPiano.getKey(keynumber+1).isRecorded() == false) return keynumber+1;
236 }
237 // Search for incomplete key from the left
238 for (int k=0; k<K; ++k) {
239 if (mPiano.getKey(k).isRecorded() == false) {
240 return k;
241 }
242 }
243 return -1;
244}
