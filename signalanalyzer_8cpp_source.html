105//-----------------------------------------------------------------------------
106// Message receiver and dispatcher

113
114void SignalAnalyzer::handleMessage(MessagePtr m)
115{
116 switch (m->getType())
117 {
118 case Message::MSG_PROJECT_FILE:
119 {
120 // stop the thread
121 this->stop();
122 auto mpf(std::static_pointer_cast<MessageProjectFile>(m));
123 mPiano = &mpf->getPiano();
124 updateOverpull();
125 break;
126 }
127 case Message::MSG_RECORDING_STARTED: // start thread
128 start();
129 break;
130 case Message::MSG_RECORDING_ENDED:
131 mRecording = false; // set a flag to terminate thread
132 break;
133 case Message::MSG_KEY_SELECTION_CHANGED:
134 {
135 auto mfkr(std::static_pointer_cast<MessageKeySelectionChanged>(m));
136 mSelectedKey = mfkr->getKeyNumber();
137 mKeyForced = mfkr->isForced();
138 updateDataBufferSize();
139 break;
140 }
141 case Message::MSG_MODE_CHANGED:
142 {
143 auto mmc(std::static_pointer_cast<MessageModeChanged>(m));
144 switch (mmc->getMode())
145 {
146 case MODE_RECORDING:
147 changeRole(ROLE_RECORD_KEYSTROKE);
148 break;
149 case MODE_TUNING:
150 changeRole(ROLE_ROLLING_FFT);
151 updateOverpull();
152 break;
153 default:
154 changeRole(ROLE_IDLE);
155 break;
156 }
157
158 break;
159 }
160 default:
161 break;
162 }
163}

198
199void SignalAnalyzer::updateDataBufferSize()
200{
201 std::lock_guard<std::mutex> lock(mDataBufferMutex);
202
203 switch (mAnalyzerRole.load())
204 {
205 case ROLE_IDLE:
206 break;
207 case ROLE_RECORD_KEYSTROKE:
208 {
209 // Initialize the local circular buffer which holds about a minute of data
210 mDataBuffer.resize(mAudioRecorder->getSampleRate() *
211 AUDIO_BUFFER_SIZE_IN_SECONDS);
212 break;
213 }
214 case ROLE_ROLLING_FFT:
215 {
216 // Initialize the local circular buffer which holds 0.5...3 seconds of data
217 const int globalKey = mSelectedKey + 48 - mPiano->getKeyboard().getKeyNumberOfA4();
218 const double timeAtHighest = 0.5;
219 const double timeAtLowest = 3;
220 const double time = (timeAtHighest - timeAtLowest) * globalKey / 88 + timeAtLowest;
221 mDataBuffer.resize(static_cast<size_t>(mAudioRecorder->getSampleRate() * time));
222 break;
223 }
224 }
225 mDataBuffer.clear();
226}

269//-----------------------------------------------------------------------------
270// Record signal
271//-----------------------------------------------------------------------------
272
276
277void SignalAnalyzer::recordSignal()
278{
279 if (cancelThread()) {return;}
280
281 mRecording = true;
282
283 mDataBufferMutex.lock();
284 mDataBuffer.clear();
285 mDataBufferMutex.unlock();
286
287 // Reset the statistics for the majority of recognized keys
288 mKeyCountStatistics.clear();
289
290 // Create a shared pointer to a vector containing the powerspectrum
291 mPowerspectrum = std::make_shared<FFTData>();
292 EptAssert(mPowerspectrum, "powerspectrum is accessed after while loop, be sure it is a valid pointer initially");
293
294 // Define a timer in order to send only a limited rate of FFTs.
295 Timer timer;
296
297 // define the packed to store audio data
298 AudioRecorder::PacketType packet;
299
300 // read all data from the audio recorder to clear all buffered data
301 mAudioRecorder->readAll(packet);
302
303 // get the sampling rate
304 uint samplingrate = mAudioRecorder->getSampleRate();
305
306 // Loop that continuously reads the audio stream and performs FFTs
307 while (mRecording and not cancelThread())
308 {
309 timer.reset();
310 mAudioRecorder->readAll(packet); // Read audio data
311 if (packet.size() > 0)
312 {
313 // lock the data puffer if new data available during compile comutation run
314 std::lock_guard<std::mutex> lock(mDataBufferMutex);
315
316 for (auto &d : packet) mDataBuffer.push_back(d);
317
318 if (mAnalyzerRole == ROLE_RECORD_KEYSTROKE) {
319 if (mDataBuffer.size() == mDataBuffer.maximum_size()) {
320 LogW("Audio buffer size in SignalAnalyzer reached.");
321 }
322 }
323
324 // If the buffer has accumulated a certain minimum of data
325 if (mDataBuffer.size() > (samplingrate * MINIMAL_FFT_INTERVAL_IN_MILLISECONDS) / 1000)
326 {
327
328 // Get audio data and make it suitable for analysis
329 mProprocessedSignal = mDataBuffer.getOrderedData();
330
331 // check if there is data in the buffer
332 bool dataInBuffer = false;
333 for (auto &d : mProprocessedSignal) {
334 if (d * d > 0) {
335 // this element contains data
336 dataInBuffer = true;
337 break;
338 }
339 }
340 if (!dataInBuffer) {// no data in buffer
341 continue;
342 }
343
344 // preprocess signal
345 signalPreprocessing(mProprocessedSignal);
346 CHECK_CANCEL_THREAD;
347
348 // process signal
349 signalProcessing(mProprocessedSignal, samplingrate);
350
351 // Finally let's wait until a certain minimal time has elapsed.
352 timer.waitUntil (MINIMAL_FFT_INTERVAL_IN_MILLISECONDS);
353 }
354 }
355 timer.wait(50);
356 }
357 LogI("Recording complete, total FFT size = %d.",static_cast<int>(mPowerspectrum->fft.size()));
358}

361//-----------------------------------------------------------------------------
362// Analyze final signal
363//-----------------------------------------------------------------------------
364
368
369void SignalAnalyzer::analyzeSignal()
370{
371 CHECK_CANCEL_THREAD;
372
373 // and determine key from the key statistics
374 int keynumber = identifySelectedKey();
375 LogI("Detected key: %d", keynumber);
376
377 // check the audio signal for possible
378 // clipping effects and unusually long strings of zero amplitudes
379 detectClipping(mDataBuffer.getOrderedData());
380
381 CHECK_CANCEL_THREAD;
382
383 // if the key is forced, we use the forced key number
384 if (mKeyForced) {keynumber = mSelectedKey;}
385
386 // check if a correct key was found
387 if (keynumber < 0)
388 {
389 LogI("Final key could not be found. Cancel analysis.");
390 return;
391 }
392
393 // check if found key equates the keynumber
394 if (keynumber != mSelectedKey)
395 {
396 ++mInvalidRecoringCounter;
397 LogW("Final detected key does not match the selected key. %d invalid recordings. Cancel analysis.", mInvalidRecoringCounter);
398 return;
399 }
400
401 mInvalidRecoringCounter = 0;
402
403 // If the key was successfully identified start call the FFTAnalyzer
404 if (mAnalyzerRole == ROLE_RECORD_KEYSTROKE)
405 {
406 // returns a pair consisting of error code and key-shared-ptr
407 auto result = mFFTAnalyser.analyse(mPiano, mPowerspectrum, keynumber);
408
409 if (result.first != FFTAnalyzerErrorTypes::ERR_NONE) // if error
410 MessageHandler::send<MessageNewFFTCalculated>(result.first);
411 else
412 {
413 // send the result
414 MessageHandler::send<MessageFinalKey>(keynumber, result.second);
415 MessageHandler::send<MessagePreliminaryKey>(-1,0);
416 }
417 }
418 else if (mAnalyzerRole == ROLE_ROLLING_FFT)
419 {
420 std::shared_ptr<Key> key = std::make_shared<Key>(mPiano->getKey(keynumber));
421 FrequencyDetectionResult result = mFFTAnalyser.detectFrequencyOfKnownKey(mPowerspectrum, mPiano, *key, keynumber);
422
423 if (result->error != FFTAnalyzerErrorTypes::ERR_NONE) // if error
424 MessageHandler::send<MessageNewFFTCalculated>(result->error);
425 else
426 {
427 key->setTunedFrequency(result->detectedFrequency);
428 MessageHandler::send<MessageFinalKey>(keynumber, key);
429 }
430 result->overpullInCents = key->getOverpull();
431 MessageHandler::send<MessageTuningDeviation>(result);
432 }
433}

473//-----------------------------------------------------------------------------
474// Update overpulls
475//-----------------------------------------------------------------------------
476
484
485void SignalAnalyzer::updateOverpull ()
486{
487 int K = mPiano->getKeyboard().getNumberOfKeys();
488 for (int keynumber=0; keynumber<K; ++keynumber)
489 {
490 // Compute the new overpull value depending on the current tune
491 double overpull = mOverpull.getOverpull(keynumber,mPiano);
492
493 // Get the currently displayed overpull value and compute the change
494 double currentoverpull = mPiano->getKey(keynumber).getOverpull();
495 double change = overpull-currentoverpull;
496
497 // If more change than 1/10 cents then
498 if (fabs(change) >= 0.1 or (currentoverpull!=0 and overpull==0))
499 {
500 // set new overpull value
501 std::shared_ptr<Key> key = std::make_shared<Key>(mPiano->getKey(keynumber));
502 double existingtune = key->getTunedFrequency();
503 if (existingtune>20 and currentoverpull!=0)
504 key->setTunedFrequency(existingtune*pow(2,change/1200.0));
505 key->setOverpull(overpull); // this informs the tuning curve window
506 MessageHandler::send<MessageFinalKey>(keynumber, key);
507 }
508 }

512//-----------------------------------------------------------------------------
513// Signal preprocessing
514//-----------------------------------------------------------------------------
515
528
529double SignalAnalyzer::signalPreprocessing(FFTWVector &signal)
530{
531 if (signal.size() == 0) {
532 LogW("Empty signal. Cancelling the signal preprocessing");
533 return 0 ;
534 }
535
536 const uint sr = mAudioRecorder->getSampleRate();
537 uint N=(uint)signal.size();
538
539 // 1. Remove dc-Bias and cut subsonic waves
540 // For a derivation see Mathematica file in the doc folder
541 const double dcBias = std::accumulate(signal.begin(), signal.end(), 0.0) / N;
542 //for (auto &s : signal) {s -= dcBias;}
543
544 const double f0 = 5; // Frequency to be suppressed by 50%
545 const double a=10.8828*f0/sr; // Damping factor
546 double follow=0;
547 for (auto &s : signal)
548 {
549 // remove dc-bias
550 s -= dcBias;
551
552 // subsonic waves
553 follow += a*(s-follow);
554 s -= follow;
555 }
556
557 // cut silence
558 mAudioRecorder->cutSilence(signal);
559
560 // signal size may be changed
561 N=(uint)signal.size();
562
563 // 2. Determine the initial energy of the keystroke:
564 uint blocksize = std::min(N,sr/5); // 0.2 sec
565 if (blocksize<=0) return 0;
566 double E0=0;
567 for (uint i=0; i<blocksize; i++) E0+=signal[i]*signal[i];
568 E0 *= 2.0/blocksize;
569
570 if (mAnalyzerRole == ROLE_RECORD_KEYSTROKE)
571 {
572 // 3. Modify the input signal in such a way that the volume is constant
573 const double gamma=50.0/sr;
574 double E1=E0,E2=E0,E3=E0;
575 for (auto &s : signal) {
576 E1 += gamma * (s*s-E1);
577 E2 += gamma * (E1-E2);
578 E3 += gamma * (E2-E3);
579 s /= (sqrt(fabs(E3))+0.001);
580 }
581
582 // 4. Fade in and out at the end of the buffer
583 blocksize = N/50;
584 for (uint i=0; i<blocksize; i++) {
585 signal[i] *= (double)i/blocksize;
586 signal[N-i-1] *= (double)i/blocksize;
587 }
588 return N/log(E0/E3)/sr; // return the average decay time of the envelope
589 }
590 else if (mAnalyzerRole == ROLE_ROLLING_FFT)
591 {
592 // 4. Apply a hanning window
593 for (size_t n = 0; n < N; ++n) {
594// signal[n] *= 0.5 * (1.0 - cos((MathTools::TWO_PI * n) / (N - 1)));
595 }
596
597 }
598
599 return 0;
600}

602//-----------------------------------------------------------------------------
603// Signal preprocessing
604//-----------------------------------------------------------------------------
605
614
615void SignalAnalyzer::signalProcessing(FFTWVector &signal, int samplingrate) {
616 if (signal.size() == 0) {
617 LogW("Empty signal. Cancelling the signal processing");
618 return;
619 } else if (samplingrate <= 0) {
620 LogW("Invalid sampling rate. Cancelling the signal processing");
621 return;
622 }
623
624 mPowerspectrum = std::make_shared<FFTData>();
625 mPowerspectrum->samplingRate = samplingrate;
626 PerformFFT(signal, mPowerspectrum->fft);
627 if (cancelThread()) return;
628
629 // The FFT is too long to be plotted. Therefore, we
630 // create here a shorter polygon and transmit it by a message
631 std::shared_ptr<FFTPolygon> polygon = std::make_shared<FFTPolygon>();
632 createPolygon (mPowerspectrum->fft, *polygon.get());
633
634 MessageHandler::send<MessageNewFFTCalculated>
635 ((!mRecording) ? MessageNewFFTCalculated::FFTMessageTypes::FinalFFT :
636 MessageNewFFTCalculated::FFTMessageTypes::NewFFT,
637 mPowerspectrum, polygon);
638
639 // recognize key
640 mKeyRecognizer.recognizeKey(false, mPiano, mPowerspectrum, mSelectedKey, mKeyForced);
641
642 if (mAnalyzerRole == ROLE_ROLLING_FFT) {
643 analyzeSignal();
644 }
645
646 // automatic key selection
647 if (!Settings::getSingleton().isAutomaticKeySelectionDisabled()) {
648 // if the recognizer finds a key next to the current key, change the selected key
649 // of course if the user forced the key, this should not work
650 // also select the key if there is non selected
651 int identifiedKey = identifySelectedKey();
652 if ((!mKeyForced && abs(identifiedKey - mSelectedKey) == 1 && identifiedKey != -1)
653 || mSelectedKey == -1) {
654 MessageHandler::sendUnique<MessageKeySelectionChanged>(identifiedKey, mPiano->getKeyPtr(identifiedKey));
655 }
656 }
657}

683//-----------------------------------------------------------------------------
684// Clipping detector
685//-----------------------------------------------------------------------------
686
700
701bool SignalAnalyzer::detectClipping(FFTWVector signal)
702{
703 int nullcnt=0, maxcnt=0, mincnt=0;
704 double maxamp=0, minamp=0;
705 for (auto &y : signal)
706 {
707 if (y>maxamp) maxamp=y;
708 else if (y>=maxamp*0.99) maxcnt++;
709 if (y<minamp) minamp=y;
710 else if (y<=minamp*0.99) mincnt++;
711 if (y==0) nullcnt++;
712 }
713 const int threshold = static_cast<int>(signal.size()) / 50;
714 if (maxcnt+mincnt > threshold)
715 {
716 LogW("SignalAnalyzer: High-amplitude clipping detected");
717 return true;
718 }
719 else if (nullcnt>threshold)
720 {
721 LogW("SignalAnalyzer: Highly intermittent signal detected (lot of zero amplitudes)");
722 return true;
723 }
724 return false;
725}

728//-----------------------------------------------------------------------------
729// Create polygon for drawing
730//-----------------------------------------------------------------------------
731
738
739void SignalAnalyzer::createPolygon (const FFTWVector &powerspec, FFTPolygon &poly) const
740{
741 const double fmin = 25;
742 const double fmax = 6000;
743 const double cents = 10;
744 const double factor = pow(2.0,cents/2400);
745
746 size_t fftsize = powerspec.size();
747 EptAssert(fftsize>0,"powerspectum has to be non-empty");
748 int samplingrate = mAudioRecorder->getSampleRate();
749 auto q = [fftsize,samplingrate] (double f) { return 2*fftsize*f/samplingrate; };
750
751 double qs1 = q(fmin/factor);
752 int q1 = std::max<int>(0, MathTools::roundToInteger(qs1));
753 double leftarea = (q1-qs1+0.5)*powerspec[q1];
754 double ymax=0, df = samplingrate / 2 / fftsize;
755 for (double f=fmin; f<=fmax; f=std::max(f*factor*factor,f+df))
756 {
757 const double qs2 = q(f*factor);
758 const int q2 = std::min<int>(MathTools::roundToInteger(qs2), static_cast<int>(powerspec.size()) - 1);
759 double sum=0;
760 for (int q=q1+1; q<=q2; ++q) sum += powerspec[q];
761 const double rightarea = (q2-qs2+0.5)*powerspec[q2];
762 const double y = sum + leftarea - rightarea;
763 if (y>ymax) ymax=y;
764 poly[f] = y;
765 q1=q2; qs1=qs2; leftarea=rightarea;
766 }
767 if (ymax <= 0) {
768 LogW("Power should be nonzero, possibly empty data.");
769 } else {
770 for (auto &p : poly) p.second /= ymax; // normalize
771 }
772}

775//-----------------------------------------------------------------------------
776// Identify the selected key
777//-----------------------------------------------------------------------------
778
788
789int SignalAnalyzer::identifySelectedKey()
790{
791 // Take the majority selection from KeyRecognizer
792 std::lock_guard<std::mutex> lock(mKeyCountStatisticsMutex);
793 if (mKeyCountStatistics.size()==0) return -1;
794 auto cmp = [](const std::pair<int,int>& p1, const std::pair<int,int>& p2)
795 { return p1.second < p2.second; };
796 auto max = std::max_element(mKeyCountStatistics.begin(),mKeyCountStatistics.end(),cmp);
797 if (max->second > static_cast<int>(mKeyCountStatistics.size()/2)) return max->first;
798
799 return -1;
800}

818
819void SignalAnalyzer::keyRecognized(int keyIndex, double frequency)
820{
821 EptAssert(mPiano, "Piano has to be set.");
822
823 if (mAnalyzerRole == ROLE_RECORD_KEYSTROKE)
824 {
825 // fetch the current key from the statistics
826 if (keyIndex >= 0 and keyIndex < mPiano->getKeyboard().getNumberOfKeys())
827 {
828 std::lock_guard<std::mutex> lock(mKeyCountStatisticsMutex);
829 mKeyCountStatistics[keyIndex]++;
830 }
831 MessageHandler::send<MessagePreliminaryKey>(identifySelectedKey(),frequency);
832 }
833 else
834 {
835 // In all other modes only one key is held in the statistics
836 std::lock_guard<std::mutex> lock(mKeyCountStatisticsMutex);
837 // this is the only key
838 mKeyCountStatistics.clear();
839 mKeyCountStatistics[keyIndex]++;
840 // send it as a preliminary recognized key
841 MessageHandler::send<MessagePreliminaryKey>(keyIndex,frequency);
842 }
